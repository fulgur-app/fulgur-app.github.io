20/// Handle ureq errors and convert them to SynchronizationError with appropriate logging
21///
22/// ### Description
23/// Centralizes ureq error handling logic that was duplicated across sync modules.
24/// Maps all ureq error variants to appropriate SynchronizationError types and logs
25/// the error with context.
26///
27/// ### Arguments
28/// - `error`: The ureq error to handle
29/// - `context`: Human-readable context for logging (e.g., "Failed to get devices")
30///
31/// ### Returns
32/// - `SynchronizationError`: The mapped synchronization error
33pub fn handle_ureq_error(error: ureq::Error, context: &str) -> SynchronizationError {
34    match error {
35        ureq::Error::StatusCode(code) => {
36            log::error!("{}: HTTP status {}", context, code);
37            if code == 401 || code == 403 {
38                SynchronizationError::AuthenticationFailed
39            } else if code == 400 {
40                SynchronizationError::BadRequest
41            } else {
42                SynchronizationError::ServerError(code)
43            }
44        }
45        ureq::Error::Io(io_error) => {
46            log::error!("{} (IO): {}", context, io_error);
47            match io_error.kind() {
48                std::io::ErrorKind::ConnectionRefused => SynchronizationError::ConnectionFailed,
49                std::io::ErrorKind::TimedOut => SynchronizationError::Timeout(io_error.to_string()),
50                std::io::ErrorKind::ConnectionReset | std::io::ErrorKind::ConnectionAborted => {
51                    SynchronizationError::ConnectionFailed
52                }
53                std::io::ErrorKind::AddrNotAvailable => SynchronizationError::HostNotFound,
54                _ => SynchronizationError::Other(io_error.to_string()),
55            }
56        }
57        ureq::Error::ConnectionFailed => {
58            log::error!("{}: Connection failed", context);
59            SynchronizationError::ConnectionFailed
60        }
61        ureq::Error::HostNotFound => {
62            log::error!("{}: Host not found", context);
63            SynchronizationError::HostNotFound
64        }
65        ureq::Error::Timeout(timeout) => {
66            log::error!("{}: Timeout ({})", context, timeout);
67            SynchronizationError::Timeout(timeout.to_string())
68        }
69        e => {
70            log::error!("{}: {}", context, e);
71            SynchronizationError::Other(e.to_string())
72        }
73    }
74}

76/// Initial synchronization with the server. This endpoint returns both the encryption key and any shared files waiting for this device.
77///
78/// ### Arguments
79/// - `server_url`: The server URL
80/// - `email`: The email
81/// - `key`: The encrypted device authentication key
82///
83/// ### Returns
84/// - `Ok(BeginResponse)`: The begin response containing encryption key and shared files
85/// - `Err(SynchronizationError)`: If the synchronization could not be performed
86pub fn initial_synchronization(
87    synchronization_settings: &SynchronizationSettings,
88    token_state: Arc<TokenStateManager>,
89    http_agent: &ureq::Agent,
90) -> Result<BeginResponse, SynchronizationError> {
91    let Some(server_url) = synchronization_settings.server_url.clone() else {
92        return Err(SynchronizationError::ServerUrlMissing);
93    };
94    let Some(public_key) = synchronization_settings.public_key.clone() else {
95        return Err(SynchronizationError::MissingEncryptionKey); //TODO
96    };
97    let token = get_valid_token(synchronization_settings, token_state, http_agent)?;
98    let begin_url = format!("{}/api/begin", server_url);
99    let payload = InitialSynchronizationPayload { public_key };
100    let mut response = http_agent
101        .post(begin_url)
102        .header("Authorization", &format!("Bearer {}", token))
103        .send_json(payload)
104        .map_err(|e| handle_ureq_error(e, "Failed to begin synchronization"))?;
105    let body = match response.body_mut().read_to_string() {
106        Ok(body) => body,
107        Err(e) => {
108            log::error!("Failed to read response body: {}", e);
109            return Err(SynchronizationError::Other(e.to_string()));
110        }
111    };
112    let begin_response: BeginResponse = match serde_json::from_str(&body) {
113        Ok(response) => response,
114        Err(e) => {
115            log::error!("Failed to parse response body: {}", e);
116            return Err(SynchronizationError::InvalidResponse(e.to_string()));
117        }
118    };
119    log::info!(
120        "Initial synchronization successful with {} shared files",
121        begin_response.shares.len()
122    );
123    Ok(begin_response)
124}

126/// Fetches shared files from the server and stores them for processing without blocking app startup
127///
128/// ### Arguments
129/// - `entity`: The Fulgur entity
130/// - `cx`: The application context
131pub fn begin_synchronization(entity: &gpui::Entity<crate::fulgur::Fulgur>, cx: &gpui::App) {
132    if !entity
133        .read(cx)
134        .settings
135        .app_settings
136        .synchronization_settings
137        .is_synchronization_activated
138    {
139        return;
140    }
141    let shared = cx.global::<crate::fulgur::shared_state::SharedAppState>();
142    if shared
143        .sync_state
144        .initialized
145        .swap(true, std::sync::atomic::Ordering::SeqCst)
146    {
147        log::debug!("Sync already initialized by another window");
148        return;
149    }
150    log::info!("Initializing sync system");
151    let settings = entity.read(cx).settings.clone();
152    let sync_server_connection_status = shared.sync_state.connection_status.clone();
153    let pending_shared_files = shared.sync_state.pending_shared_files.clone();
154    let device_name = shared.sync_state.device_name.clone();
155    let sse_tx = entity.read(cx).sse_state.sse_event_tx.clone();
156    let sse_shutdown_flag = entity.read(cx).sse_state.sse_shutdown_flag.clone();
157    let token_state = shared.sync_state.token_state.clone();
158    let http_agent = Arc::clone(&shared.http_agent);
159    thread::spawn(move || {
160        // Small delay to ensure app initialization doesn't block
161        thread::sleep(Duration::from_millis(100));
162        let server_url = settings
163            .app_settings
164            .synchronization_settings
165            .server_url
166            .clone();
167        let email = settings.app_settings.synchronization_settings.email.clone();
168        let key = match load_device_api_key_from_keychain() {
169            Ok(value) => value,
170            Err(_) => {
171                set_sync_server_connection_status(
172                    sync_server_connection_status.clone(),
173                    SynchronizationStatus::Disconnected,
174                );
175                return;
176            }
177        };
178        if server_url.is_none() || email.is_none() || key.is_none() {
179            set_sync_server_connection_status(
180                sync_server_connection_status.clone(),
181                SynchronizationStatus::Disconnected,
182            );
183            return;
184        }
185        match initial_synchronization(
186            &settings.app_settings.synchronization_settings,
187            Arc::clone(&token_state),
188            &http_agent,
189        ) {
190            Ok(begin_response) => {
191                log::info!("Successfully connected to sync server");
192                set_sync_server_connection_status(
193                    sync_server_connection_status.clone(),
194                    SynchronizationStatus::Connected,
195                );
196                {
197                    let mut device_name = device_name.lock();
198                    *device_name = Some(begin_response.device_name);
199                }
200                {
201                    let mut files = pending_shared_files.lock();
202                    *files = begin_response
203                        .shares
204                        .into_iter()
205                        .map(|mut share| {
206                            share.file_name = sanitize_filename(&share.file_name);
207                            share
208                        })
209                        .collect();
210                }
211                if let (Some(tx), Some(shutdown)) = (sse_tx, sse_shutdown_flag) {
212                    log::info!("Starting SSE connection for real-time updates");
213                    if let Err(e) = connect_sse(
214                        &settings.app_settings.synchronization_settings,
215                        tx,
216                        shutdown,
217                        sync_server_connection_status.clone(),
218                        Arc::clone(&token_state),
219                        Arc::clone(&http_agent),
220                    ) {
221                        log::error!("Failed to start SSE connection: {}", e);
222                    }
223                } else {
224                    log::warn!(
225                        "SSE event sender or shutdown flag not available, cannot start SSE connection"
226                    );
227                }
228            }
229            Err(e) => {
230                log::error!("Failed to fetch shared files: {}", e);
231                set_sync_server_connection_status(
232                    sync_server_connection_status,
233                    SynchronizationStatus::Disconnected,
234                );
235            }
236        }
237    });
238}

252/// Perform initial synchronization with the server
253///
254/// ### Arguments
255/// - `entity`: The Fulgur entity
256/// - `cx`: The context
257///
258/// ### Returns
259/// - `SynchronizationStatus`: The status of the connection to the sync server
260pub fn perform_initial_synchronization(
261    entity: Entity<crate::fulgur::Fulgur>,
262    cx: &mut App,
263) -> SynchronizationStatus {
264    let synchronization_settings = entity
265        .read(cx)
266        .settings
267        .app_settings
268        .synchronization_settings
269        .clone();
270    let shared = cx.global::<crate::fulgur::shared_state::SharedAppState>();
271    let token_state = Arc::clone(&shared.sync_state.token_state);
272    let http_agent = &shared.http_agent;
273    let result = initial_synchronization(&synchronization_settings, token_state, http_agent);
274    let (notification, sync_server_connection_status) = match result {
275        Ok(begin_response) => {
276            let shared = cx.global::<crate::fulgur::shared_state::SharedAppState>();
277            {
278                let mut name = shared.sync_state.device_name.lock();
279                *name = Some(begin_response.device_name.clone());
280            }
281            {
282                let mut files = shared.sync_state.pending_shared_files.lock();
283                *files = begin_response.shares;
284            }
285            (
286                (
287                    NotificationType::Success,
288                    SharedString::from(format!(
289                        "Connection successful as {}",
290                        begin_response.device_name
291                    )),
292                ),
293                SynchronizationStatus::Connected,
294            )
295        }
296        Err(e) => (
297            (
298                NotificationType::Error,
299                SharedString::from(format!("Connection failed: {}", e)),
300            ),
301            SynchronizationStatus::from_error(&e),
302        ),
303    };
304    entity.update(cx, |this, cx| {
305        let shared = cx.global::<crate::fulgur::shared_state::SharedAppState>();
306        set_sync_server_connection_status(
307            shared.sync_state.connection_status.clone(),
308            sync_server_connection_status,
309        );
310        // Store notification to be displayed on next render
311        this.pending_notification = Some(notification);
312    });
313    sync_server_connection_status
314}

340impl fmt::Display for SynchronizationError {
341    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
342        match self {
343            SynchronizationError::AuthenticationFailed => write!(f, "Authentication failed"),
344            SynchronizationError::BadRequest => write!(f, "Bad request"),
345            SynchronizationError::CompressionFailed => write!(f, "Compression failed"),
346            SynchronizationError::ConnectionFailed => write!(f, "Cannot connect to sync server"),
347            SynchronizationError::ContentMissing => write!(f, "Content is missing"),
348            SynchronizationError::ContentTooLarge => write!(f, "Content is too large to share"),
349            SynchronizationError::DeviceIdsMissing => write!(f, "Device IDs are missing"),
350            SynchronizationError::DeviceKeyMissing => write!(f, "Key is missing"),
351            SynchronizationError::EmailMissing => write!(f, "Email is missing"),
352            SynchronizationError::EncryptionFailed => write!(f, "Encryption failed"),
353            SynchronizationError::FileNameMissing => write!(f, "File name is missing"),
354            SynchronizationError::HostNotFound => write!(f, "Host not found"),
355            SynchronizationError::InvalidResponse(e) => write!(f, "{}", e),
356            SynchronizationError::MissingEncryptionKey => write!(f, "Missing encryption key"),
357            SynchronizationError::MissingExpirationDate => write!(f, "Missing expiration date"),
358            SynchronizationError::MissingPublicKey(e) => {
359                write!(f, "Missing public key for device: {}", e)
360            }
361            SynchronizationError::Other(e) => write!(f, "{}", e),
362            SynchronizationError::ServerError(e) => write!(f, "{}", e),
363            SynchronizationError::ServerUrlMissing => write!(f, "Server URL is missing"),
364            SynchronizationError::Timeout(timeout) => write!(f, "Timeout: {}", timeout),
365        }
366    }

370pub enum SynchronizationStatus {
371    Connected,
372    Disconnected,
373    AuthenticationFailed,
374    ConnectionFailed,
375    Other,
376    NotActivated,

379impl SynchronizationStatus {
380    /// Convert the error to a synchronization status
381    ///
382    /// ### Arguments
383    /// - `error`: The error
384    ///
385    /// ### Returns
386    /// - `SynchronizationStatus`: The synchronization status
387    pub fn from_error(error: &SynchronizationError) -> SynchronizationStatus {
388        match error {
389            SynchronizationError::AuthenticationFailed => {
390                SynchronizationStatus::AuthenticationFailed
391            }
392            SynchronizationError::HostNotFound => SynchronizationStatus::ConnectionFailed,
393            SynchronizationError::ConnectionFailed => SynchronizationStatus::ConnectionFailed,
394            SynchronizationError::Timeout(_) => SynchronizationStatus::ConnectionFailed,
395            _ => SynchronizationStatus::Other,
396        }
397    }
398
399    /// Check if the synchronization status is connected
400    ///
401    /// ### Returns
402    /// - `true` if the synchronization status is connected, `false` otherwise
403    pub fn is_connected(&self) -> bool {
404        match self {
405            SynchronizationStatus::Connected => true,
406            SynchronizationStatus::Disconnected => false,
407            SynchronizationStatus::AuthenticationFailed => false,
408            SynchronizationStatus::ConnectionFailed => false,
409            SynchronizationStatus::Other => false,
410            SynchronizationStatus::NotActivated => false,
411        }
412    }
413}

415impl Fulgur {
416    /// Process shared files from the sync server
417    ///
418    /// ### Arguments
419    /// - `window`: The window to create new tabs in
420    /// - `cx`: The application context
421    pub fn process_shared_files_from_sync(&mut self, window: &mut Window, cx: &mut Context<Self>) {
422        let shared_files_to_open = if let Some(mut pending) = self
423            .shared_state(cx)
424            .sync_state
425            .pending_shared_files
426            .try_lock()
427        {
428            if pending.is_empty() {
429                Vec::new()
430            } else {
431                log::info!(
432                    "Processing {} shared file(s) from sync server",
433                    pending.len()
434                );
435                pending.drain(..).collect()
436            }
437        } else {
438            Vec::new()
439        };
440        if !shared_files_to_open.is_empty() {
441            let encryption_key_opt = match load_private_key_from_keychain() {
442                Ok(key) => key,
443                Err(_) => {
444                    log::error!("Cannot decrypt shared files: encryption key not available");
445                    None
446                }
447            };
448            if let Some(encryption_key) = encryption_key_opt {
449                for shared_file in shared_files_to_open {
450                    if shared_file.content.len() > share::MAX_SYNC_SHARE_PAYLOAD_BYTES {
451                        log::warn!(
452                            "Skipping shared file '{}' from device {}: encrypted payload exceeds {} bytes",
453                            shared_file.file_name,
454                            shared_file.source_device_id,
455                            share::MAX_SYNC_SHARE_PAYLOAD_BYTES
456                        );
457                        continue;
458                    }
459                    let decrypted_result =
460                        crypto_helper::decrypt_bytes(&shared_file.content, &encryption_key)
461                            .and_then(|compressed_bytes| {
462                                if compressed_bytes.len() > share::MAX_SYNC_SHARE_PAYLOAD_BYTES {
463                                    return Err(anyhow::anyhow!(
464                                        "Compressed payload exceeds {} bytes limit",
465                                        share::MAX_SYNC_SHARE_PAYLOAD_BYTES
466                                    ));
467                                }
468                                share::decompress_content(&compressed_bytes)
469                            });
470                    match decrypted_result {
471                        Ok(decrypted_content) => {
472                            let tab_id = self.next_tab_id;
473                            self.next_tab_id += 1;
474                            let new_tab = Tab::Editor(editor_tab::EditorTab::from_content(
475                                tab_id,
476                                decrypted_content,
477                                shared_file.file_name.clone(),
478                                window,
479                                cx,
480                                &self.settings.editor_settings,
481                            ));
482                            self.tabs.push(new_tab);
483                            self.active_tab_index = Some(self.tabs.len() - 1);
484                            log::info!("Opened shared file: {}", shared_file.file_name);
485                        }
486                        Err(e) => {
487                            log::error!(
488                                "Failed to decrypt shared file {}: {}",
489                                shared_file.file_name,
490                                e
491                            );
492                        }
493                    }
494                }
495            } else {
496                log::error!("Cannot decrypt shared files: encryption key not available");
497            }
498        }
499    }
