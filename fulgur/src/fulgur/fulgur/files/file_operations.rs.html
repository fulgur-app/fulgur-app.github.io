18/// Detect encoding from file bytes
19///
20/// ### Arguments
21/// - `bytes`: The bytes to detect encoding from
22///
23/// ### Returns
24/// - `(String, String)`: The detected encoding and decoded string
25pub fn detect_encoding_and_decode(bytes: &[u8]) -> (String, String) {
26    if let Ok(text) = std::str::from_utf8(bytes) {
27        log::debug!("File encoding detected as UTF-8");
28        return (UTF_8.to_string(), text.to_string());
29    }
30    let mut detector = EncodingDetector::new();
31    detector.feed(bytes, true);
32    let encoding = detector.guess(None, true);
33    let (decoded, _, had_errors) = encoding.decode(bytes);
34    let encoding_name = if had_errors {
35        match std::str::from_utf8(bytes) {
36            Ok(text) => {
37                log::debug!("File encoding detected as UTF-8 (after error recovery)");
38                return (UTF_8.to_string(), text.to_string());
39            }
40            Err(_) => {
41                let text = String::from_utf8_lossy(bytes).to_string();
42                log::warn!("File encoding detection failed, using UTF-8 lossy conversion");
43                return (UTF_8.to_string(), text);
44            }
45        }
46    } else {
47        encoding.name().to_string()
48    };
49    log::debug!("File encoding detected as: {}", encoding_name);
50    (encoding_name, decoded.to_string())
51}

54    /// Find the index of a tab with the given file path
55    ///
56    /// ### Arguments
57    /// - `path`: The path to search for
58    ///
59    /// ### Returns
60    /// - `Some(usize)`: The index of the tab if found
61    /// - `None`: If the tab was not found
62    pub fn find_tab_by_path(&self, path: &PathBuf) -> Option<usize> {
63        self.tabs.iter().position(|tab| {
64            if let Tab::Editor(editor_tab) = tab {
65                if let Some(ref tab_path) = editor_tab.file_path {
66                    tab_path == path
67                } else {
68                    false
69                }
70            } else {
71                false
72            }
73        })
74    }

76    /// Reload tab content from disk
77    ///
78    /// ### Arguments
79    /// - `tab_index`: The index of the tab to reload
80    /// - `window`: The window context
81    /// - `cx`: The application context
82    pub fn reload_tab_from_disk(
83        &mut self,
84        tab_index: usize,
85        window: &mut Window,
86        cx: &mut Context<Self>,
87    ) {
88        let path = if let Some(Tab::Editor(editor_tab)) = self.tabs.get(tab_index) {
89            editor_tab.file_path.clone()
90        } else {
91            None
92        };
93        if let Some(path) = path {
94            log::debug!("Reloading tab content from disk: {:?}", path);
95            match std::fs::read(&path) {
96                Ok(bytes) => {
97                    let (encoding, contents) = detect_encoding_and_decode(&bytes);
98                    if let Some(Tab::Editor(editor_tab)) = self.tabs.get_mut(tab_index) {
99                        editor_tab.content.update(cx, |input_state, cx| {
100                            input_state.set_value(&contents, window, cx);
101                        });
102                        editor_tab.original_content = contents;
103                        editor_tab.encoding = encoding;
104                        editor_tab.modified = false;
105                        editor_tab.update_file_tooltip_cache(bytes.len());
106                        editor_tab.update_language(window, cx, &self.settings.editor_settings);
107                        log::debug!("Tab reloaded successfully from disk: {:?}", path);
108                    }
109                }
110                Err(e) => {
111                    log::error!("Failed to reload file {:?}: {}", path, e);
112                }
113            }
114        }
115    }

117    /// Internal helper function to open a file from a path. This function handles reading the file, detecting encoding, and creating the editor tab
118    ///
119    /// ### Arguments
120    /// - `view`: The view entity (WeakEntity)
121    /// - `window`: The async window context
122    /// - `path`: The path to the file to open
123    ///
124    /// ### Returns
125    /// - `None`: If the file could not be opened
126    /// - `Some(())`: If the file was opened successfully
127    async fn open_file_from_path(
128        view: WeakEntity<Self>,
129        window: &mut AsyncWindowContext,
130        path: PathBuf,
131    ) -> Option<()> {
132        log::debug!("Attempting to open file: {:?}", path);
133        let bytes = match std::fs::read(&path) {
134            Ok(bytes) => {
135                log::debug!("Successfully read file: {:?} ({} bytes)", path, bytes.len());
136                bytes
137            }
138            Err(e) => {
139                log::error!("Failed to read file {:?}: {}", path, e);
140                return None;
141            }
142        };
143        let (encoding, contents) = detect_encoding_and_decode(&bytes);
144        window
145            .update(|window, cx| {
146                _ = view.update(cx, |this, cx| {
147                    let editor_tab = EditorTab::from_file(
148                        FromFileParams {
149                            id: this.next_tab_id,
150                            path: path.clone(),
151                            contents,
152                            encoding,
153                            is_modified: false,
154                        },
155                        window,
156                        cx,
157                        &this.settings.editor_settings,
158                    );
159                    this.tabs.push(Tab::Editor(editor_tab));
160                    this.active_tab_index = Some(this.tabs.len() - 1);
161                    this.next_tab_id += 1;
162                    this.watch_file(&path);
163                    this.focus_active_tab(window, cx);
164                    if let Err(e) = this.settings.add_file(path.clone()) {
165                        log::error!("Failed to add file to recent files: {}", e);
166                    }
167                    let shared = this.shared_state(cx);
168                    let update_info = shared.update_info.lock().clone();
169                    let update_link = if let Some(info) = update_info {
170                        Some(info.download_url.clone())
171                    } else {
172                        None
173                    };
174                    let menus = menus::build_menus(&this.settings.get_recent_files(), update_link);
175                    cx.set_menus(menus);
176                    let title = path
177                        .file_name()
178                        .map(|file_name| file_name.to_string_lossy().to_string());
179                    this.set_title(title, cx);
180                    log::debug!("File opened successfully in new tab: {:?}", path);
181                    if let Err(e) = this.save_state(cx, window) {
182                        log::error!("Failed to save app state after opening file: {}", e);
183                        this.pending_notification = Some((
184                            NotificationType::Warning,
185                            format!("File opened but failed to save state: {}", e).into(),
186                        ));
187                    }
188                    cx.notify();
189                });
190            })
191            .ok();
192        Some(())
193    }

195    /// Open a file
196    ///
197    /// ### Arguments
198    /// - `window`: The window to open the file in
199    /// - `cx`: The application context
200    pub fn open_file(&mut self, window: &mut Window, cx: &mut Context<Self>) {
201        let path_future = cx.prompt_for_paths(PathPromptOptions {
202            files: true,
203            directories: false,
204            multiple: false,
205            prompt: None,
206        });
207        cx.spawn_in(window, async move |view, window| {
208            let paths = path_future.await.ok()?.ok()??;
209            let path = paths.first()?.clone();
210
211            // Check if tab already exists for this path
212            let should_open_new = window
213                .update(|window, cx| {
214                    view.update(cx, |this, cx| {
215                        if let Some(tab_index) = this.find_tab_by_path(&path) {
216                            log::debug!(
217                                "Tab already exists for {:?} at index {}, focusing and reloading",
218                                path,
219                                tab_index
220                            );
221                            if let Some(Tab::Editor(editor_tab)) = this.tabs.get(tab_index) {
222                                if editor_tab.modified {
223                                    log::debug!("Tab is modified, reloading content from disk");
224                                    this.reload_tab_from_disk(tab_index, window, cx);
225                                } else {
226                                    log::debug!("Tab is not modified, just focusing it");
227                                }
228                            }
229                            this.active_tab_index = Some(tab_index);
230                            this.focus_active_tab(window, cx);
231                            cx.notify();
232                            false // Don't open new tab
233                        } else {
234                            true // Open new tab
235                        }
236                    })
237                    .ok()
238                })
239                .ok()??;
240
241            if should_open_new {
242                Self::open_file_from_path(view, window, path).await
243            } else {
244                Some(())
245            }
246        })
247        .detach();
248    }

250    /// Open a file from a given path
251    ///
252    /// ### Behavior
253    /// First detects if the file is already open, and will focus on that tab if that's the case.
254    ///
255    /// ### Arguments
256    /// - `window`: The window to open the file in
257    /// - `cx`: The application context
258    /// - `path`: The path to the file to open
259    pub fn do_open_file(&mut self, window: &mut Window, cx: &mut Context<Self>, path: PathBuf) {
260        if let Some(tab_index) = self.find_tab_by_path(&path) {
261            log::debug!(
262                "Tab already exists for {:?} at index {}, focusing and reloading if modified",
263                path,
264                tab_index
265            );
266            if let Some(Tab::Editor(editor_tab)) = self.tabs.get(tab_index) {
267                if editor_tab.modified {
268                    log::debug!("Tab is modified, reloading content from disk");
269                    self.reload_tab_from_disk(tab_index, window, cx);
270                } else {
271                    log::debug!("Tab is not modified, just focusing it");
272                }
273            }
274            self.active_tab_index = Some(tab_index);
275            self.focus_active_tab(window, cx);
276            cx.notify();
277            return;
278        }
279        let window_manager = cx.global::<crate::fulgur::window_manager::WindowManager>();
280        if let Some(existing_window_id) =
281            window_manager.find_window_with_file(&path, self.window_id, cx)
282        {
283            let file_name = path
284                .file_name()
285                .and_then(|n| n.to_str())
286                .unwrap_or("Unknown file");
287            let message = format!("File '{}' is already open in another window", file_name);
288            window.push_notification((NotificationType::Info, SharedString::from(message)), cx);
289            log::debug!(
290                "File {:?} is already open in window {:?}",
291                path,
292                existing_window_id
293            );
294            return;
295        }
296        cx.spawn_in(window, async move |view, window| {
297            Self::open_file_from_path(view, window, path).await
298        })
299        .detach();
300    }

323    /// Save a file
324    ///
325    /// ### Arguments
326    /// - `window`: The window to save the file in
327    /// - `cx`: The application context
328    pub fn save_file(&mut self, window: &mut Window, cx: &mut Context<Self>) {
329        if self.tabs.is_empty() {
330            return;
331        }
332        let Some(active_tab_index) = self.active_tab_index else {
333            return;
334        };
335        let active_tab = &self.tabs[active_tab_index];
336        let (path, content_entity) = match active_tab {
337            Tab::Editor(editor_tab) => {
338                let Some(file_path) = editor_tab.file_path.clone() else {
339                    self.save_file_as(window, cx);
340                    return;
341                };
342                (file_path, editor_tab.content.clone())
343            }
344            Tab::Settings(_) => return,
345        };
346        let contents = content_entity.read(cx).text().to_string();
347        log::debug!("Saving file: {:?} ({} bytes)", path, contents.len());
348        if let Err(e) = atomic_write_file(&path, contents.as_bytes()) {
349            log::error!("Failed to save file {:?}: {}", path, e);
350            let file_name = path.file_name().and_then(|n| n.to_str()).unwrap_or("file");
351            window.push_notification(
352                (
353                    NotificationType::Error,
354                    SharedString::from(format!("Failed to save '{}': {}", file_name, e)),
355                ),
356                cx,
357            );
358            return;
359        }
360        log::debug!("File saved successfully: {:?}", path);
361        self.file_watch_state
362            .last_file_saves
363            .insert(path.clone(), std::time::Instant::now());
364        if let Tab::Editor(editor_tab) = &mut self.tabs[active_tab_index] {
365            editor_tab.mark_as_saved(cx);
366            editor_tab.update_file_tooltip_cache(contents.len());
367        }
368        cx.notify();
369    }

371    /// Save a file as
372    ///
373    /// ### Arguments
374    /// - `window`: The window to save the file as in
375    /// - `cx`: The application context
376    pub fn save_file_as(&mut self, window: &mut Window, cx: &mut Context<Self>) {
377        if self.tabs.is_empty() {
378            return;
379        }
380        let Some(active_tab_index) = self.active_tab_index else {
381            return;
382        };
383        let (content_entity, directory, suggested_filename) = match &self.tabs[active_tab_index] {
384            Tab::Editor(editor_tab) => {
385                let dir = if let Some(ref path) = editor_tab.file_path {
386                    path.parent()
387                        .unwrap_or(std::path::Path::new("."))
388                        .to_path_buf()
389                } else {
390                    std::env::current_dir().unwrap_or_default()
391                };
392                let suggested = editor_tab.get_suggested_filename();
393                (editor_tab.content.clone(), dir, suggested)
394            }
395            Tab::Settings(_) => return,
396        };
397        let path_future = cx.prompt_for_new_path(&directory, suggested_filename.as_deref());
398        cx.spawn_in(window, async move |view, window| {
399            let path = path_future.await.ok()?.ok()??;
400            let contents = window
401                .update(|_, cx| content_entity.read(cx).text().to_string())
402                .ok()?;
403            log::debug!("Saving file as: {:?} ({} bytes)", path, contents.len());
404            if let Err(e) = atomic_write_file(&path, contents.as_bytes()) {
405                log::error!("Failed to save file {:?}: {}", path, e);
406                let file_name = path.file_name().and_then(|n| n.to_str()).unwrap_or("file");
407                let message = SharedString::from(format!("Failed to save '{}': {}", file_name, e));
408                window
409                    .update(|_, cx| {
410                        _ = view.update(cx, |this, cx| {
411                            this.pending_notification = Some((NotificationType::Error, message));
412                            cx.notify();
413                        });
414                    })
415                    .ok()?;
416                return None;
417            }
418            log::debug!("File saved successfully as: {:?}", path);
419            window
420                .update(|window, cx| {
421                    _ = view.update(cx, |this, cx| {
422                        let old_path = if let Some(Tab::Editor(editor_tab)) =
423                            this.tabs.get(active_tab_index)
424                        {
425                            editor_tab.file_path.clone()
426                        } else {
427                            None
428                        };
429                        if let Some(old_path) = old_path {
430                            this.unwatch_file(&old_path);
431                        }
432                        this.file_watch_state
433                            .last_file_saves
434                            .insert(path.clone(), std::time::Instant::now());
435                        if let Some(Tab::Editor(editor_tab)) = this.tabs.get_mut(active_tab_index) {
436                            editor_tab.file_path = Some(path.clone());
437                            editor_tab.title = path
438                                .file_name()
439                                .and_then(|n| n.to_str())
440                                .unwrap_or(UNTITLED)
441                                .to_string()
442                                .into();
443                            editor_tab.mark_as_saved(cx);
444                            editor_tab.update_file_tooltip_cache(contents.len());
445                            editor_tab.update_language(window, cx, &this.settings.editor_settings);
446                            cx.notify();
447                        }
448                        this.watch_file(&path);
449                    });
450                })
451                .ok()?;
452            Some(())
453        })
454        .detach();
455    }

511    /// Process pending files from macOS "Open With" events
512    ///
513    /// ### Arguments
514    /// - `window`: The window to open files in
515    /// - `cx`: The application context
516    pub fn process_pending_files_from_macos(
517        &mut self,
518        window: &mut Window,
519        cx: &mut Context<Self>,
520    ) {
521        let shared = self.shared_state(cx);
522        let should_process_files = cx
523            .global::<window_manager::WindowManager>()
524            .get_last_focused()
525            .map(|id| id == self.window_id)
526            .unwrap_or(true); // If no last focused window, allow this one to process
527        let files_to_open = if should_process_files {
528            if let Some(mut pending) = shared.pending_files_from_macos.try_lock() {
529                if pending.is_empty() {
530                    Vec::new()
531                } else {
532                    log::info!(
533                        "Processing {} pending file(s) from macOS open event in window {:?}",
534                        pending.len(),
535                        self.window_id
536                    );
537                    pending.drain(..).collect()
538                }
539            } else {
540                Vec::new()
541            }
542        } else {
543            Vec::new()
544        };
545        for file_path in files_to_open {
546            self.handle_open_file_from_cli(window, cx, file_path);
547        }
548    }

550    /// Update search results if the search query has changed
551    ///
552    /// ### Arguments
553    /// - `window`: The window containing the search bar and editor
554    /// - `cx`: The application context
555    pub fn update_search_if_needed(&mut self, window: &mut Window, cx: &mut Context<Self>) {
556        if self.search_state.show_search {
557            let current_query = self.search_state.search_input.read(cx).text().to_string();
558            if current_query != self.search_state.last_search_query {
559                self.search_state.last_search_query = current_query;
560                self.perform_search(window, cx);
561                // Restore focus to the search input after perform_search
562                let search_focus = self.search_state.search_input.read(cx).focus_handle(cx);
563                window.focus(&search_focus);
564            }
565        }
566    }
