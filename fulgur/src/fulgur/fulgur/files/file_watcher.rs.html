49pub enum FileWatchEvent {

51    Deleted(PathBuf),
52    Renamed { from: PathBuf, to: PathBuf },
53    Error(String),

56pub struct FileWatcher {

58    watched_paths: HashMap<PathBuf, SystemTime>,

62impl FileWatcher {
63    /// Creates a new file watcher
64    ///
65    /// ### Returns
66    /// - `(FileWatcher, Receiver<FileWatchEvent>)`: A tuple containing the file watcher and the event receiver
67    ///   - `FileWatcher`: The file watcher instance
68    ///   - `Receiver<FileWatchEvent>`: The event receiver to receive the file watch events from the file watcher
69    pub fn new() -> (Self, Receiver<FileWatchEvent>) {
70        let (event_tx, event_rx) = channel();
71        let watcher = Self {
72            watcher: None,
73            watched_paths: HashMap::new(),
74            event_tx,
75        };
76        (watcher, event_rx)
77    }
78
79    /// Starts the file watcher
80    ///
81    /// ### Returns
82    /// - `Ok(())`: If the file watcher was started successfully
83    /// - `Err(NotifyError)`: If the file watcher could not be started
84    pub fn start(&mut self) -> Result<(), NotifyError> {
85        if self.watcher.is_some() {
86            return Ok(());
87        }
88        let event_tx = self.event_tx.clone();
89        let watcher =
90            notify::recommended_watcher(move |res: Result<Event, NotifyError>| match res {
91                Ok(event) => {
92                    Self::handle_notify_event(event, &event_tx);
93                }
94                Err(e) => {
95                    let _ = event_tx.send(FileWatchEvent::Error(e.to_string()));
96                }
97            })?;
98        self.watcher = Some(watcher);
99        Ok(())
100    }
101
102    /// Handles a notify event and converts it to a FileWatchEvent
103    ///
104    /// ### Description
105    /// - If the event is a modification, it sends a Modified event to the event sender
106    /// - If the event is a deletion, it sends a Deleted event to the event sender
107    /// - If the event is a creation after a rename or save, it sends a Modified event to the event sender
108    /// - If the event is other, it ignores it
109    /// - Ignore other event types (access, etc.)
110    ///
111    /// ### Arguments
112    /// - `event`: The notify event to handle
113    /// - `event_tx`: The event sender to send the events to
114    fn handle_notify_event(event: Event, event_tx: &Sender<FileWatchEvent>) {
115        use notify::event::ModifyKind;
116
117        match event.kind {
118            EventKind::Modify(ModifyKind::Name(_)) => {
119                if event.paths.len() == 2 {
120                    let from = event.paths[0].clone();
121                    let to = event.paths[1].clone();
122                    let _ = event_tx.send(FileWatchEvent::Renamed { from, to });
123                }
124            }
125            EventKind::Modify(_) => {
126                for path in event.paths {
127                    let _ = event_tx.send(FileWatchEvent::Modified(path));
128                }
129            }
130            EventKind::Remove(_) => {
131                for path in event.paths {
132                    let _ = event_tx.send(FileWatchEvent::Deleted(path));
133                }
134            }
135            EventKind::Create(_) => {
136                for path in event.paths {
137                    let _ = event_tx.send(FileWatchEvent::Modified(path));
138                }
139            }
140            _ => {}
141        }
142    }

144    /// Starts watching a file
145    ///
146    /// ### Arguments
147    /// - `path`: The path to the file to watch
148    ///
149    /// ### Returns
150    /// - `Ok(())`: If the file was watched successfully
151    /// - `Err(String)`: If the file could not be watched
152    pub fn watch_file(&mut self, path: PathBuf) -> Result<(), String> {
153        if self.watched_paths.contains_key(&path) {
154            return Ok(());
155        }
156        if self.watcher.is_none() {
157            self.start().map_err(|e| e.to_string())?;
158        }
159        let modified_time = std::fs::metadata(&path)
160            .and_then(|m| m.modified())
161            .unwrap_or(SystemTime::UNIX_EPOCH);
162        if let Some(watcher) = &mut self.watcher {
163            watcher
164                .watch(&path, RecursiveMode::NonRecursive)
165                .map_err(|e| format!("Failed to watch file {}: {}", path.display(), e))?;
166        }
167        self.watched_paths.insert(path.clone(), modified_time);
168        log::debug!("Started watching file: {}", path.display());
169        Ok(())
170    }
171
172    /// Stops watching a file
173    ///
174    /// ### Arguments
175    /// - `path`: The path to the file to stop watching
176    pub fn unwatch_file(&mut self, path: &PathBuf) {
177        if !self.watched_paths.contains_key(path) {
178            return;
179        }
180        if let Some(watcher) = &mut self.watcher
181            && let Err(e) = watcher.unwatch(path)
182        {
183            log::warn!("Failed to unwatch file {}: {}", path.display(), e);
184        }
185
186        self.watched_paths.remove(path);
187        log::debug!("Stopped watching file: {}", path.display());
188    }
189
190    /// Stops the file watcher completely
191    pub fn stop(&mut self) {
192        if let Some(mut watcher) = self.watcher.take() {
193            for path in self.watched_paths.keys() {
194                let _ = watcher.unwatch(path);
195            }
196        }
197        self.watched_paths.clear();
198        log::debug!("File watcher stopped");
199    }
200}

209impl Fulgur {
210    /// Handle file watch events received from the file watcher
211    ///
212    /// ### Description
213    /// - If the event is a modification, it shows a conflict dialog if the file is modified and the tab is active
214    /// - If the event is a deletion, it shows a notification that the file was deleted
215    /// - If the event is a rename, it shows a notification that the file was renamed
216    /// - If the event is an error, it logs the error
217    ///
218    /// ### Arguments
219    /// - `event`: The file watch event to handle
220    /// - `window`: The window context
221    /// - `cx`: The application context
222    pub fn handle_file_watch_event(
223        &mut self,
224        event: FileWatchEvent,
225        window: &mut Window,
226        cx: &mut Context<Self>,
227    ) {
228        match event {
229            FileWatchEvent::Modified(path) => {
230                let now = Instant::now();
231                if let Some(&last_time) = self.file_watch_state.last_file_events.get(&path)
232                    && now.duration_since(last_time) < Duration::from_millis(500)
233                {
234                    return;
235                }
236                if let Some(&save_time) = self.file_watch_state.last_file_saves.get(&path)
237                    && now.duration_since(save_time) < Duration::from_millis(500)
238                {
239                    return;
240                }
241                self.file_watch_state
242                    .last_file_events
243                    .insert(path.clone(), now);
244                if let Some(tab_index) = self.find_tab_by_path(&path)
245                    && let Some(Tab::Editor(editor_tab)) = self.tabs.get(tab_index)
246                {
247                    if editor_tab.modified {
248                        let is_active = self.active_tab_index == Some(tab_index);
249
250                        if is_active {
251                            self.show_file_conflict_dialog(path, tab_index, window, cx);
252                        } else {
253                            self.file_watch_state
254                                .pending_conflicts
255                                .insert(path, tab_index);
256                        }
257                    } else {
258                        self.reload_tab_from_disk(tab_index, window, cx);
259                        self.show_notification_file_reloaded(&path, window, cx);
260                    }
261                }
262            }
263            FileWatchEvent::Deleted(path) => {
264                self.show_notification_file_deleted(&path, window, cx);
265            }
266            FileWatchEvent::Renamed { from, to } => {
267                if let Some(tab_index) = self.find_tab_by_path(&from) {
268                    self.unwatch_file(&from);
269                    self.watch_file(&to);
270                    if let Some(Tab::Editor(editor_tab)) = self.tabs.get_mut(tab_index) {
271                        editor_tab.file_path = Some(to.clone());
272                        editor_tab.title = to
273                            .file_name()
274                            .and_then(|n| n.to_str())
275                            .unwrap_or("Untitled")
276                            .to_string()
277                            .into();
278                    }
279                    self.show_notification_file_renamed(&from, &to, window, cx);
280                }
281            }
282            FileWatchEvent::Error(msg) => {
283                log::error!("File watcher error: {}", msg);
284            }
285        }
286    }

288    /// Start the file watcher and watch all open files
289    pub fn start_file_watcher(&mut self) {
290        let (mut watcher, receiver) = FileWatcher::new();
291        if let Err(e) = watcher.start() {
292            log::error!("Failed to start file watcher: {}", e);
293            return;
294        }
295        for tab in &self.tabs {
296            if let Tab::Editor(editor_tab) = tab
297                && let Some(path) = &editor_tab.file_path
298                && let Err(e) = watcher.watch_file(path.clone())
299            {
300                log::warn!("Failed to watch file {}: {}", path.display(), e);
301            }
302        }
303        self.file_watch_state.file_watcher = Some(watcher);
304        self.file_watch_state.file_watch_events = Some(receiver);
305    }
306
307    /// Stop the file watcher
308    pub fn stop_file_watcher(&mut self) {
309        if let Some(mut watcher) = self.file_watch_state.file_watcher.take() {
310            watcher.stop();
311        }
312        self.file_watch_state.file_watch_events = None;
313    }

315    /// Add a file to the watcher
316    ///
317    /// ### Arguments
318    /// - `path`: The path to the file to watch
319    pub fn watch_file(&mut self, path: &std::path::Path) {
320        if let Some(watcher) = &mut self.file_watch_state.file_watcher
321            && let Err(e) = watcher.watch_file(path.to_path_buf())
322        {
323            log::warn!("Failed to watch file {}: {}", path.display(), e);
324        }
325    }

327    /// Remove a file from the watcher
328    ///
329    /// ### Arguments
330    /// - `path`: The path to the file to unwatch
331    pub fn unwatch_file(&mut self, path: &PathBuf) {
332        if let Some(watcher) = &mut self.file_watch_state.file_watcher {
333            watcher.unwatch_file(path);
334        }
335    }

337    /// Collect and process file watch events:
338    /// - Modified: File content changed externally (may trigger auto-reload or conflict dialog)
339    /// - Deleted: File was deleted externally (shows notification)
340    /// - Renamed: File was moved/renamed (updates tab path and continues watching)
341    ///
342    /// ### Arguments
343    /// - `window`: The window containing the tabs with watched files
344    /// - `cx`: The application context
345    pub fn process_file_watch_events(&mut self, window: &mut Window, cx: &mut Context<Self>) {
346        let events = collect_events(&self.file_watch_state.file_watch_events);
347        for event in events {
348            self.handle_file_watch_event(event, window, cx);
349        }
350    }
