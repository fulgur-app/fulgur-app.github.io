245pub struct Themes {
246    pub default_themes: Vec<ThemeFile>,
247    pub user_themes: Vec<ThemeFile>,
248}
249
250impl Themes {
251    /// Load the theme settings from the themes folder
252    ///
253    /// ### Returns
254    /// - `anyhow::Result<Self>`: The theme settings
255    pub fn load() -> anyhow::Result<Self> {
256        let themes_dir = themes_directory_path()?;
257        let themes_files = fs::read_dir(&themes_dir)?;
258        let default_themes: Vec<ThemeFile> = BundledThemes::iter()
259            .map(|file| ThemeFile::load(themes_dir.join(file.as_ref())))
260            .collect::<Result<Vec<ThemeFile>, anyhow::Error>>()?;
261        let default_themes_names = BundledThemes::iter()
262            .map(|file| file.as_ref().to_string())
263            .collect::<Vec<String>>();
264        let user_themes: Vec<ThemeFile> = themes_files
265            .filter_map(|entry| {
266                entry.ok().and_then(|entry| {
267                    let filename = entry.file_name().to_string_lossy().to_string();
268                    if !default_themes_names.contains(&filename) {
269                        Some(entry)
270                    } else {
271                        None
272                    }
273                })
274            })
275            .filter_map(|entry| ThemeFile::load(entry.path()).ok())
276            .collect();
277        Ok(Themes {
278            default_themes,
279            user_themes,
280        })
281    }

321    /// Save the settings to a specific path
322    ///
323    /// ### Description
324    /// Core implementation for saving settings. Can be used with custom paths
325    /// for testing or alternative storage locations. Uses file locking to prevent
326    /// corruption when multiple windows write simultaneously.
327    ///
328    /// ### Arguments
329    /// - `path`: The path to save the settings to
330    ///
331    /// ### Returns
332    /// - `Ok(())`: The result of the operation
333    /// - `Err(anyhow::Error)`: If there was an error saving the settings
334    pub fn save_to_path(&mut self, path: &PathBuf) -> anyhow::Result<()> {
335        // Serialize settings to JSON first (fast, no I/O)
336        let json = serde_json::to_string_pretty(&self)?;
337
338        let file = fs::OpenOptions::new()
339            .read(true)
340            .write(true)
341            .create(true)
342            .truncate(false)
343            .open(path)
344            .map_err(|e| anyhow::anyhow!("Failed to open settings file: {}", e))?;
345        file.lock_exclusive()
346            .map_err(|e| anyhow::anyhow!("Failed to acquire lock on settings file: {}", e))?;
347
348        atomic_write_file(path, json.as_bytes())
349    }
350
351    /// Load the settings from a specific path
352    ///
353    /// ### Description
354    /// Core implementation for loading settings. Can be used with custom paths
355    /// for testing or alternative storage locations. Uses shared file locking to
356    /// allow concurrent reads while preventing reads during writes.
357    ///
358    /// ### Arguments
359    /// - `path`: The path to load the settings from
360    ///
361    /// ### Returns
362    /// - `Ok(Settings)`: The loaded settings
363    /// - `Err(anyhow::Error)`: If there was an error loading the settings
364    pub fn load_from_path(path: &PathBuf) -> anyhow::Result<Self> {
365        let file = fs::OpenOptions::new()
366            .read(true)
367            .open(path)
368            .map_err(|e| anyhow::anyhow!("Failed to open settings file for reading: {}", e))?;
369        file.lock_shared().map_err(|e| {
370            anyhow::anyhow!("Failed to acquire shared lock on settings file: {}", e)
371        })?;
372        let mut reader = std::io::BufReader::new(&file);
373        let mut json = String::new();
374        std::io::Read::read_to_string(&mut reader, &mut json)
375            .map_err(|e| anyhow::anyhow!("Failed to read settings: {}", e))?;
376        let settings: Settings = serde_json::from_str(&json)
377            .map_err(|e| anyhow::anyhow!("Failed to parse settings: {}", e))?;
378        Ok(settings)
379    }

428impl Fulgur {
429    /// Update settings and propagate to all windows
430    ///
431    /// This method should be called whenever settings are changed. It will:
432    /// 1. Save settings to disk
433    /// 2. Update shared settings in SharedAppState
434    /// 3. Increment the shared settings version (so other windows detect the change)
435    /// 4. Set settings_changed flag for this window
436    /// 5. Force all windows to re-render immediately
437    ///
438    /// ### Arguments
439    /// - `cx`: The application context
440    ///
441    /// ### Returns
442    /// - `anyhow::Result<()>`: Result of the operation
443    pub fn update_and_propagate_settings(&mut self, cx: &mut Context<Self>) -> anyhow::Result<()> {
444        // Save settings to disk
445        if let Err(e) = self.settings.save() {
446            log::error!("Failed to save settings: {}", e);
447            self.pending_notification = Some((
448                NotificationType::Error,
449                format!("Failed to save settings: {}", e).into(),
450            ));
451            return Err(e);
452        }
453
454        // Update shared settings
455        let shared = self.shared_state(cx);
456        *shared.settings.lock() = self.settings.clone();
457
458        // Increment the version counter so other windows detect the change
459        let new_version = shared
460            .settings_version
461            .fetch_add(1, std::sync::atomic::Ordering::SeqCst)
462            + 1;
463        self.local_settings_version = new_version;
464
465        // Mark settings as changed for this window
466        self.settings_changed = true;
467
468        log::debug!(
469            "Window {:?} updated settings to version {}, notifying other windows",
470            self.window_id,
471            new_version
472        );
473
474        // Force other windows to re-render immediately
475        // (Skip the current window to avoid reentrancy issues - it will re-render naturally)
476        let current_window_id = self.window_id;
477        let window_manager = cx.global::<window_manager::WindowManager>();
478        let all_windows = window_manager.get_all_windows();
479
480        // Defer notifications to avoid reentrancy issues
481        cx.defer(move |cx| {
482            for weak_window in all_windows.iter() {
483                if let Some(window_entity) = weak_window.upgrade() {
484                    // Skip the current window (already updating)
485                    let should_notify = window_entity.read(cx).window_id != current_window_id;
486                    if should_notify {
487                        window_entity.update(cx, |_, cx| {
488                            cx.notify();
489                        });
490                    }
491                }
492            }
493        });
494
495        Ok(())
496    }
497
498    /// Synchronize settings from other windows
499    ///
500    /// ### Arguments
501    /// - `cx`: The application context
502    pub fn synchronize_settings_from_other_windows(&mut self, cx: &mut Context<Self>) {
503        let shared = self.shared_state(cx);
504        let shared_version = shared
505            .settings_version
506            .load(std::sync::atomic::Ordering::Relaxed);
507        if shared_version > self.local_settings_version {
508            // Settings have been updated in another window - reload them
509            let shared_settings = shared.settings.lock().clone();
510            self.settings = shared_settings;
511            self.local_settings_version = shared_version;
512            self.settings_changed = true;
513            log::debug!(
514                "Window {:?} detected settings change from another window (version {} -> {})",
515                self.window_id,
516                self.local_settings_version,
517                shared_version
518            );
519        }
520    }
521
522    /// Propagate settings changes to tabs
523    ///
524    /// ### Arguments
525    /// - `window`: The window containing the tabs
526    /// - `cx`: The application context
527    pub fn propagate_settings_to_tabs(&mut self, window: &mut Window, cx: &mut Context<Self>) {
528        if !self.tabs_pending_update.is_empty() {
529            let settings = self.settings.editor_settings.clone();
530            for tab_index in self.tabs_pending_update.drain() {
531                if let Some(Tab::Editor(editor_tab)) = self.tabs.get_mut(tab_index) {
532                    editor_tab.update_settings(window, cx, &settings);
533                }
534            }
535        }
536        if self.settings_changed {
537            let settings = self.settings.editor_settings.clone();
538            for tab_index in self.rendered_tabs.iter().copied().collect::<Vec<_>>() {
539                if let Some(Tab::Editor(editor_tab)) = self.tabs.get_mut(tab_index) {
540                    editor_tab.update_settings(window, cx, &settings);
541                }
542            }
543            self.settings_changed = false;
544        }
545    }
546
547    /// Track newly rendered tabs and mark them for settings update
548    ///
549    /// ### Arguments
550    /// - `cx`: The application context
551    pub fn track_newly_rendered_tabs(&mut self, cx: &mut Context<Self>) {
552        if let Some(index) = self.active_tab_index {
553            let is_newly_rendered = !self.rendered_tabs.contains(&index);
554            self.rendered_tabs.insert(index);
555            if is_newly_rendered {
556                self.tabs_pending_update.insert(index);
557                cx.notify();
558            }
559        }
560    }
