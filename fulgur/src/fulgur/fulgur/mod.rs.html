121    /// Create a new Fulgur instance
122    ///
123    /// ### Arguments
124    /// - `window`: The window to create the Fulgur instance in
125    /// - `cx`: The application context
126    /// - `window_index`: Index of this window in saved state (0 = first window, etc.). Use usize::MAX for new empty windows.
127    ///
128    /// ### Returns
129    /// - `Entity<Self>`: The new Fulgur instance
130    pub fn new(window: &mut Window, cx: &mut App, window_index: usize) -> Entity<Self> {
131        let title_bar = CustomTitleBar::new(window, cx);
132        let shared = cx.global::<shared_state::SharedAppState>();
133        let settings = shared.settings.lock().clone();
134        let window_id = WindowId::default();
135        let search_input = cx.new(|cx| InputState::new(window, cx).placeholder("Search"));
136        let replace_input = cx.new(|cx| InputState::new(window, cx).placeholder("Replace"));
137        let jump_to_line_input =
138            cx.new(|cx| InputState::new(window, cx).placeholder("Jump to line or line:character"));
139        let entity = cx.new(|cx| {
140            let search_subscription =
141                cx.subscribe(&search_input, |this: &mut Self, _, ev: &InputEvent, cx| {
142                    if let InputEvent::Change = ev
143                        && this.search_state.show_search
144                    {
145                        cx.notify();
146                    }
147                });
148            Self {
149                window_id,
150                focus_handle: cx.focus_handle(),
151                title_bar,
152                tabs: vec![],
153                active_tab_index: None,
154                next_tab_id: 0,
155                search_state: SearchState::new(search_input, replace_input, search_subscription),
156                jump_to_line_input,
157                pending_jump: None,
158                jump_to_line_dialog_open: false,
159                settings,
160                settings_changed: false,
161                local_settings_version: 0,
162                rendered_tabs: HashSet::new(),
163                tabs_pending_update: HashSet::new(),
164                tab_scroll_handle: ScrollHandle::new(),
165                pending_tab_scroll: None,
166                file_watch_state: FileWatchState::new(),
167                sse_state: SseState::new(),
168                pending_notification: None,
169                cached_window_bounds: None,
170            }
171        });
172        let (sse_tx, sse_rx) = std::sync::mpsc::channel();
173        let sse_shutdown_flag = Arc::new(AtomicBool::new(false));
174        entity.update(cx, |this, cx| {
175            this.sse_state.sse_events = Some(sse_rx);
176            this.sse_state.sse_event_tx = Some(sse_tx);
177            this.sse_state.sse_shutdown_flag = Some(sse_shutdown_flag);
178            let shared = cx.global::<shared_state::SharedAppState>();
179            if let Some(error_msg) = shared.sync_error.lock().as_ref() {
180                this.pending_notification =
181                    Some((NotificationType::Error, error_msg.clone().into()));
182            }
183            if window_index == usize::MAX {
184                let initial_tab = Tab::Editor(EditorTab::new(
185                    0,
186                    crate::fulgur::ui::components_utils::UNTITLED,
187                    window,
188                    cx,
189                    &this.settings.editor_settings,
190                ));
191                this.tabs.push(initial_tab);
192                this.active_tab_index = Some(0);
193                this.next_tab_id = 1;
194            } else {
195                this.load_state(window, cx, window_index);
196                this.pending_tab_scroll = this.active_tab_index;
197            }
198            if this.settings.editor_settings.watch_files {
199                this.start_file_watcher();
200            }
201        });
202        sync::synchronization::begin_synchronization(&entity, cx);
203        entity
204    }

278    /// ### Arguments

280    /// - `cx`: The application context
281    fn process_update_notifications(&mut self, window: &mut Window, cx: &mut Context<Self>) {
282        let update_info = {
283            let shared = self.shared_state(cx);
284            shared.update_info.lock().take()

347        {
348            app_content = app_content.child(self.render_status_bar(cx));
349        }
350        app_content
351    }

353    /// Assemble the final UI tree with all layers
354    ///
355    /// ### Arguments
356    /// - `app_content`: The main content area (from `build_app_content_with_actions()`)
357    /// - `window`: The window to assemble the UI for
358    /// - `cx`: The application context
359    ///
360    /// ### Returns
361    /// - `impl IntoElement`: The complete UI tree ready to be rendered
362    fn assemble_ui_tree(
363        &self,
364        app_content: impl IntoElement,
365        window: &mut Window,
366        cx: &mut Context<Self>,
367    ) -> impl IntoElement {
368        // Create root layout: TitleBar OUTSIDE of focus-tracked content
369        // This is critical for Windows hit-testing to work!
370        let root_content = v_flex()
371            .size_full()
372            .child(self.title_bar.clone())
373            .child(app_content);
374        div()
375            .size_full()
376            .child(root_content)
377            .children(Root::render_sheet_layer(window, cx))
378            .children(Root::render_notification_layer(window, cx))
379            .children(Root::render_dialog_layer(window, cx))
380    }

396impl Render for Fulgur {
397    /// Render the Fulgur instance
398    ///
399    /// ### Arguments
400    /// - `window`: The window to render the Fulgur instance in
401    /// - `cx`: The application context
402    ///
403    /// ### Returns
404    /// - `impl IntoElement`: The rendered Fulgur instance
405    fn render(&mut self, window: &mut Window, cx: &mut Context<Self>) -> impl IntoElement {
406        self.process_window_state_updates(window, cx);
407        self.process_update_notifications(window, cx);
408        self.synchronize_settings_from_other_windows(cx);
409        self.process_pending_files_from_macos(window, cx);
410        self.process_shared_files_from_sync(window, cx);
411        self.process_file_watch_events(window, cx);
412        self.process_sse_events(window, cx);
413        if self.tabs.is_empty() {
414            self.active_tab_index = None;
415        }
416        self.update_search_if_needed(window, cx);
417        self.propagate_settings_to_tabs(window, cx);
418        self.track_newly_rendered_tabs(cx);
419        self.handle_pending_jump_to_line(window, cx);
420        if !self.jump_to_line_dialog_open {
421            window.close_dialog(cx);
422            self.jump_to_line_dialog_open = true;
423        }
424        self.update_modified_status(cx);
425        self.process_pending_tab_scroll(cx);
426        let active_tab = self
427            .active_tab_index
428            .and_then(|index| self.tabs.get(index).cloned());
429        let app_content = self.build_app_content_with_actions(active_tab.clone(), window, cx);
430        self.assemble_ui_tree(app_content, window, cx)
431    }

434impl Fulgur {
435    /// Process a deferred scroll-to-tab request
436    ///
437    /// GPUI's ScrollHandle needs one render cycle to populate child bounds and overflow
438    /// state before `scroll_to_item` can work. On the first frame, layout hasn't happened
439    /// yet so the scroll is silently dropped. This method waits until child bounds are
440    /// available (meaning layout has completed at least once), then issues the scroll.
441    ///
442    /// ### Arguments
443    /// - `cx`: The application context
444    fn process_pending_tab_scroll(&mut self, cx: &mut Context<Self>) {
445        if let Some(index) = self.pending_tab_scroll {
446            if self.tab_scroll_handle.bounds_for_item(0).is_some() {
447                self.tab_scroll_handle.scroll_to_item(index);
448                self.pending_tab_scroll = None;
449            } else {
450                cx.notify();
451            }
452        }
453    }
454
455    /// Render the content area (editor or settings)
456    ///
457    /// ### Arguments
458    /// - `active_tab`: The active tab (if any)
459    /// - `window`: The window context
460    /// - `cx`: The application context
461    ///
462    /// ### Returns
463    /// - `AnyElement`: The rendered content area element (wrapped in AnyElement)
464    fn render_content_area(
465        &self,
466        active_tab: Option<Tab>,
467        window: &mut Window,
468        cx: &mut Context<Self>,
469    ) -> AnyElement {
470        if let Some(tab) = active_tab {
471            match tab {
472                Tab::Editor(editor_tab) => {
473                    let editor_input = Input::new(&editor_tab.content)
474                        .bordered(false)
475                        .p_0()
476                        .h_full()
477                        .font_family("Monaco")
478                        .text_size(px(self.settings.editor_settings.font_size))
479                        .focus_bordered(false);
480                    if editor_tab.language == SupportedLanguage::Markdown
481                        && editor_tab.show_markdown_preview
482                    {
483                        return v_flex()
484                            .w_full()
485                            .flex_1()
486                            .child(
487                                h_resizable("markdown-preview-container")
488                                    .child(resizable_panel().child(
489                                        div().id("markdown-editor").size_full().child(editor_input),
490                                    ))
491                                    .child(
492                                        resizable_panel().child(
493                                            TextView::markdown(
494                                                "markdown-preview",
495                                                editor_tab.content.read(cx).value().clone(),
496                                                window,
497                                                cx,
498                                            )
499                                            .flex_none()
500                                            .py_0()
501                                            .px_2()
502                                            .scrollable(true)
503                                            .selectable(true)
504                                            .bg(cx.theme().muted),
505                                        ),
506                                    ),
507                            )
508                            .into_any_element();
509                    }
510                    return v_flex()
511                        .w_full()
512                        .flex_1()
513                        .child(editor_input)
514                        .into_any_element();
515                }
516                Tab::Settings(_) => {
517                    return v_flex()
518                        .id("settings-tab-scrollable")
519                        .w_full()
520                        .flex_1()
521                        .overflow_y_scrollbar()
522                        .child(self.render_settings(window, cx))
523                        .into_any_element();
524                }
525            }
526        }
527        v_flex().w_full().flex_1().into_any_element()
528    }
