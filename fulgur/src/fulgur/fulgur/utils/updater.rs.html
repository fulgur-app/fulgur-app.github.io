36/// Parse a version string into a Version struct
37///
38/// ### Arguments
39/// - `version_str`: The version string to parse
40///
41/// ### Returns
42/// - `Ok(Version)`: The parsed version
43/// - `Err(anyhow::Error)`: If the version string could not be parsed
44fn parse_version(version_str: &str) -> anyhow::Result<Version> {
45    let cleaned = version_str.trim_start_matches('v');
46    Version::parse(cleaned).map_err(|e| e.into())
47}

49/// Check for updates
50///
51/// ### Arguments
52/// - `current_version`: The current version of the application
53///
54/// ### Returns
55/// - `Ok(Some(UpdateInfo))`: The update information if an update is available
56/// - `Ok(None)`: If no update is available
57/// - `Err(anyhow::Error)`: If the update check could not be performed
58pub fn check_for_updates(current_version: String) -> anyhow::Result<Option<UpdateInfo>> {
59    let mut response = ureq::get(GITHUB_API_URL)
60        .header("User-Agent", "Fulgur")
61        .call()?;
62    if response.status() != 200 {
63        return Err(anyhow::anyhow!("GitHub API error: {}", response.status()));
64    }
65    let release: GitHubRelease = response.body_mut().read_json()?;
66    let current = parse_version(current_version.as_str())?;
67    let latest = parse_version(&release.tag_name)?;
68    if latest > current {
69        log::info!("New version available: {} -> {}", current, latest);
70        let download_url = get_platform_download_url(&release)?;
71        Ok(Some(UpdateInfo {
72            current_version: current.to_string(),
73            latest_version: latest.to_string(),
74            is_newer: true,
75            download_url,
76            release_notes: release.body.unwrap_or_default(),
77            release_page: release.html_url,
78        }))
79    } else {
80        log::info!("Already on latest version: {}", current);
81        Ok(None)
82    }
83}

85/// Get the download URL for the platform-specific asset
86///
87/// ### Arguments
88/// - `release`: The release information
89///
90/// ### Returns
91/// - `Ok(String)`: The download URL for the platform-specific asset
92/// - `Err(anyhow::Error)`: If the download URL could not be determined
93fn get_platform_download_url(release: &GitHubRelease) -> anyhow::Result<String> {
94    let platform = std::env::consts::OS;
95    let arch = std::env::consts::ARCH;
96    let pattern = match (platform, arch) {
97        ("macos", "aarch64") => "macos-aarch64", // Apple Silicon
98        ("macos", "x86_64") => "macos-x86_64",   // Intel Mac
99        ("macos", _) => "macos-universal",       // Universal binary
100        ("windows", "x86_64") => "windows-x86_64",
101        ("windows", "aarch64") => "windows-aarch64",
102        ("linux", "x86_64") => "linux-x86_64",
103        ("linux", "aarch64") => "linux-aarch64",
104        _ => {
105            return Err(anyhow::anyhow!(
106                "Unsupported platform: {}-{}",
107                platform,
108                arch
109            ));
110        }
111    };
112    for asset in &release.assets {
113        if asset.name.contains(pattern) {
114            return Ok(asset.browser_download_url.clone());
115        }
116    }
117    Ok(release.html_url.clone())
118}
