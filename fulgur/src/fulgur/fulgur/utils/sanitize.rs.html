1/// Sanitize a filename to prevent path traversal and other security issues
2///
3/// This function:
4/// - Extracts just the filename from paths (e.g., "/path/to/file.txt" â†’ "file.txt")
5/// - Preserves leading dots (hidden files like ".gitignore")
6/// - Removes control characters and null bytes
7///
8/// ### Arguments
9/// - `filename`: The filename or path to sanitize
10///
11/// ### Returns
12/// - `String`: The sanitized filename, or "untitled" if the result is empty
13///
14/// ### Examples
15/// ```
16/// # use fulgur::fulgur::utils::sanitize::sanitize_filename;
17/// assert_eq!(sanitize_filename("../../etc/passwd"), "passwd");
18/// assert_eq!(sanitize_filename("normal.txt"), "normal.txt");
19/// assert_eq!(sanitize_filename("path/to/file.txt"), "file.txt");
20/// assert_eq!(sanitize_filename(".hidden"), ".hidden");
21/// assert_eq!(sanitize_filename(""), "untitled");
22/// ```
23pub fn sanitize_filename(filename: &str) -> String {
24    // Normalize path separators to Unix style, then split and take the last component
25    let normalized = filename.replace('\\', "/");
26    let base_name = normalized.split('/').rfind(|s| !s.is_empty()).unwrap_or("");
27
28    // Remove control characters and null bytes
29    let sanitized = base_name
30        .replace('\0', "")
31        .chars()
32        .filter(|c| !c.is_control())
33        .collect::<String>();
34
35    // If the result is empty or only whitespace, return a default name
36    if sanitized.trim().is_empty() {
37        "untitled".to_string()
38    } else {
39        sanitized
40    }
41}

47    #[test]

56        assert_eq!(sanitize_filename("../config.json"), "config.json");
57        assert_eq!(sanitize_filename("..\\windows\\system32"), "system32");

63        assert_eq!(sanitize_filename("C:\\Users\\file.txt"), "file.txt");
64        assert_eq!(
65            sanitize_filename("/absolute/path/to/document.pdf"),
66            "document.pdf"
67        );
68    }
