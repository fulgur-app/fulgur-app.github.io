3/// Add Ada language support.
4pub fn add_ada_support() {
5    LanguageRegistry::singleton().register(
6        "ada",
7        &LanguageConfig::new(
8            "ada",
9            arborium_ada::language().into(),
10            vec![],
11            ADA_HIGHLIGHTS_QUERY,
12            arborium_ada::INJECTIONS_QUERY,
13            "",
14        ),
15    );
16}
