1use arborium_clojure;
2use gpui_component::highlighter::{LanguageConfig, LanguageRegistry};

4/// Add Clojure language support.
5pub fn add_clojure_support() {
6    LanguageRegistry::singleton().register(
7        "clojure",
8        &LanguageConfig::new(
9            "clojure",
10            arborium_clojure::language().into(),
11            vec![],
12            CLOJURE_HIGHLIGHTS_QUERY,
13            arborium_clojure::INJECTIONS_QUERY,
14            arborium_clojure::LOCALS_QUERY,
15        ),
16    );
17}

19/// Custom highlights query for Clojure.
20///
21/// The upstream arborium_clojure query only covers literals (numbers, strings,
22/// booleans, keywords, comments). Because Clojure is homoiconic, special forms
23/// such as `defn`, `let`, `fn`, and `if` are indistinguishable from ordinary
24/// function calls at the grammar level — both are a `list_lit` whose first
25/// child is a `sym_lit`. A `#match?` predicate on the symbol text is required
26/// to recognise them as keywords.
27///
28/// Capture names are mapped to the names recognised by gpui-component's
29/// `HIGHLIGHT_NAMES` (`keyword`, `function`, `type`, `string.special.symbol`,
30/// etc.).
31const CLOJURE_HIGHLIGHTS_QUERY: &str = r##"
32;; ──────────────── Literals ────────────────
33
34(num_lit) @number
35
36[
37  (str_lit)
38  (char_lit)
39] @string
40
41(regex_lit) @string.regex
42
43(bool_lit) @boolean
44
45(nil_lit) @constant
46
47;; Keywords are symbols prefixed with `:` — highlight as string.special.symbol
48;; so they stand out from regular identifiers without using the type colour.
49(kwd_lit) @string.special.symbol
50
51;; ──────────────── Comments ────────────────
52
53(comment) @comment
54
55;; Discard expressions (#_ form) are treated as comments.
56(dis_expr) @comment
57
58;; ──────────────── Metadata ────────────────
59
60(meta_lit     ["^"]  @attribute)
61(old_meta_lit ["#^"] @attribute)
62
63;; ──────────────── Reader macros / quoting operators ────────────────
64
65(quoting_lit          ["'"]  @operator)
66(syn_quoting_lit      ["`"]  @operator)
67(unquoting_lit        ["~"]  @operator)
68(unquote_splicing_lit ["~@"] @operator)
69(derefing_lit         ["@"]  @operator)
70(var_quoting_lit      ["#'"] @operator)
71
72;; ──────────────── Definition name extraction ────────────────
73
74;; Capture the name (second sym_lit child) of defining forms as @function.
75;; The `@_kind` capture is used only for the predicate and produces no highlight.
76((list_lit . (sym_lit) @_kind . (sym_lit) @function)
77 (#match? @_kind "^(defn-?|defmacro|defmulti|defmethod)$"))
78
79;; Capture the name of type/record/protocol definitions as @type.
80((list_lit . (sym_lit) @_kind . (sym_lit) @type)
81 (#match? @_kind "^(defrecord|deftype|definterface|defstruct|defprotocol)$"))
82
83;; Capture the namespace name in ns / in-ns forms as @type.
84((list_lit . (sym_lit) @_kind . (sym_lit) @type)
85 (#match? @_kind "^(ns|in-ns)$"))
86
87;; ──────────────── Special forms and core macros ────────────────
88
89;; This rule must appear AFTER the name-extraction rules above so that
90;; `defn`, `defrecord`, etc. are captured both as @keyword (the symbol
91;; itself) and produce a @function / @type capture for the name that follows.
92((list_lit . (sym_lit) @keyword)
93 (#match? @keyword
94  "^(def|defn|defn-|defmacro|defmulti|defmethod|defonce|defrecord|deftype|definterface|defprotocol|defstruct|ns|in-ns|let|letfn|if|if-let|if-not|if-some|when|when-let|when-not|when-some|cond|condp|case|and|or|not|do|fn|loop|recur|throw|try|catch|finally|new|set!|quote|var|declare|require|use|import|refer|refer-clojure|load|load-file|binding|doto|with-open|with-local-vars|with-redefs|locking|monitor-enter|monitor-exit|proxy|extend|extend-type|extend-protocol|reify|->|->>|as->|some->|some->>|cond->|cond->>|dotimes|doseq|for|while|when-first|lazy-seq|lazy-cat|delay|dosync|io!)$"))
95
96;; ──────────────── Function calls ────────────────
97
98;; Any other first-position symbol in a list is treated as a function call.
99;; This rule is last so that the keyword matches above take priority for
100;; special forms.
101(list_lit . (sym_lit) @function)
102"##;
