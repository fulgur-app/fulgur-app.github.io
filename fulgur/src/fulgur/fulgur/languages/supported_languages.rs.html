218/// Get the language registry name for a SupportedLanguage.

248/// Get the Language enum from a filename or file extension.
249/// Exact filename matches (e.g. `Dockerfile`, `Makefile`) are checked first,
250/// then falls back to the file extension.
251///
252/// ### Arguments
253/// - `filename`: The file name
254///
255/// ### Returns
256/// - `SupportedLanguage`: The detected language
257pub fn language_from_filename(filename: &str) -> SupportedLanguage {
258    let lower = filename.to_lowercase();
259    let exact = match lower.as_str() {
260        "dockerfile" => Some(SupportedLanguage::Dockerfile),
261        "makefile" | "gnumakefile" => Some(SupportedLanguage::Make),
262        "gemfile" | "rakefile" | "guardfile" | "podfile" => Some(SupportedLanguage::Ruby),
263        _ => None,
264    };
265    if let Some(lang) = exact {
266        return lang;
267    }
268
269    let extension = std::path::Path::new(filename)
270        .extension()
271        .and_then(|ext| ext.to_str())
272        .unwrap_or("");
273
274    if extension == "txt" {
275        return SupportedLanguage::Plain;
276    }
277
278    // Overriding gpui-component's default language mapping.
279    if let Some(lang) = match extension {
280        "scss" => Some(SupportedLanguage::Scss),
281        _ => None,
282    } {
283        return lang;
284    }
285
286    let mut language = from_language(Language::from_str(extension));
287    if language == SupportedLanguage::Plain {
288        language = match extension {
289            "ada" | "ads" | "adb" => SupportedLanguage::Ada,
290            "astro" => SupportedLanguage::Html,
291            "clojure" | "clj" | "cljs" => SupportedLanguage::Clojure,
292            "dart" => SupportedLanguage::Dart,
293            "dockerfile" => SupportedLanguage::Dockerfile,
294            "hs" | "lhs" => SupportedLanguage::Haskell,
295            "lock" => SupportedLanguage::Toml,
296            "groovy" | "gvy" | "gy" | "gsh" => SupportedLanguage::Groovy,
297            "mjs" => SupportedLanguage::JavaScript,
298            "perl" | "pl" | "pm" | "plx" => SupportedLanguage::Perl,
299            "powershell" | "ps1" | "psm1" | "psd1" => SupportedLanguage::Powershell,
300            "php" | "php3" | "php4" | "php5" | "phhtml" => SupportedLanguage::Html,
301            "m" | "M" | "mm" => SupportedLanguage::ObjectiveC,
302            "ml" | "mli" => SupportedLanguage::Ocaml,
303            "r" | "rmd" => SupportedLanguage::R,
304            "svelte" => SupportedLanguage::Svelte,
305            "svg" => SupportedLanguage::Html,
306            "tsx" | "jsx" => SupportedLanguage::React,
307            "vue" => SupportedLanguage::Vue,
308            "xml" => SupportedLanguage::Xml,
309            _ => SupportedLanguage::Plain,
310        };
311    }
312    language
313}

377impl Fulgur {
378    /// Get the current language of the active tab
379    ///
380    /// ### Returns
381    /// - `SupportedLanguage`: The active tab's language
382    pub fn get_current_language(&self) -> SupportedLanguage {
383        match self.active_tab_index {
384            Some(index) => {
385                if let Some(editor_tab) = self.tabs[index].as_editor() {
386                    editor_tab.language
387                } else {
388                    SupportedLanguage::Plain
389                }
390            }
391            None => SupportedLanguage::Plain,
392        }
393    }
394
395    /// Check if the current active tab's language is a Markdown language
396    ///
397    /// ### Returns
398    /// - `True` if the active tab's language is a Markdown language, `False` otherwise
399    pub fn is_markdown(&self) -> bool {
400        let current_language = self.get_current_language();
401        current_language == SupportedLanguage::Markdown
402            || current_language == SupportedLanguage::MarkdownInline
403    }
404}
