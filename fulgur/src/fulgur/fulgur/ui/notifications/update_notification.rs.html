1use crate::fulgur::utils::updater::UpdateInfo;
2use gpui::*;
3use gpui_component::{button::ButtonVariants, notification::Notification};
4
5/// Create an update notification with a download button
6///
7/// ### Arguments
8/// - `update_info`: Information about the available update
9///
10/// ### Returns
11/// - `Notification`: A notification with update information and download action
12pub fn make_update_notification(update_info: &UpdateInfo) -> Notification {
13    let message = SharedString::from(format!(
14        "A new version of Fulgur is available: {}",
15        update_info.latest_version
16    ));
17    let download_url = update_info.download_url.clone();
18    gpui_component::notification::Notification::new()
19        .message(message)
20        .action(move |_, _, cx| {
21            let url = download_url.clone();
22            gpui_component::button::Button::new("download")
23                .primary()
24                .label("Download")
25                .mr_2()
26                .on_click(cx.listener(move |this, _, window, cx| {
27                    match open::that(&url) {
28                        Ok(_) => {
29                            log::debug!("Successfully opened browser for update");
30                        }
31                        Err(e) => {
32                            log::error!("Failed to open browser: {}", e);
33                        }
34                    }
35                    this.dismiss(window, cx);
36                }))
37        })
38}
