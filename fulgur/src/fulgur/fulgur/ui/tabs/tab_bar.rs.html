59    /// Get the display title for a tab, including parent folder if there are duplicates
60    ///
61    /// ### Arguments
62    /// - `index`: The index of the tab
63    /// - `tab`: The tab to get the title for
64    ///
65    /// ### Returns
66    /// - `(String, Option<String>)`: A tuple of (filename, optional parent folder)
67    fn get_tab_display_title(&self, index: usize, tab: &Tab) -> (String, Option<String>) {
68        let base_title = tab.title();
69        if let Some(editor_tab) = tab.as_editor()
70            && let Some(ref path) = editor_tab.file_path
71        {
72            let filename = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
73            let duplicate_count = self
74                .tabs
75                .iter()
76                .enumerate()
77                .filter(|(i, t)| {
78                    if *i == index {
79                        return false;
80                    }
81                    if let Some(editor) = t.as_editor()
82                        && let Some(ref other_path) = editor.file_path
83                        && let Some(other_filename) =
84                            other_path.file_name().and_then(|n| n.to_str())
85                    {
86                        return other_filename == filename;
87                    }
88                    false
89                })
90                .count();
91            if duplicate_count > 0
92                && let Some(parent) = path.parent()
93                && let Some(parent_name) = parent.file_name().and_then(|n| n.to_str())
94            {
95                return (filename.to_string(), Some(format!("../{}", parent_name)));
96            }
97
98            return (filename.to_string(), None);
99        }
100        (base_title.to_string(), None)
101    }

202    /// Handle theme switching action
203    ///
204    /// Applies the selected theme, updates settings, refreshes windows, and rebuilds menus.
205    ///
206    /// ### Arguments
207    /// - `theme_name`: The name of the theme to switch to (as SharedString from action)
208    /// - `cx`: The application context
209    pub fn switch_to_theme(&mut self, theme_name: gpui::SharedString, cx: &mut Context<Self>) {
210        if let Some(theme_config) = ThemeRegistry::global(cx)
211            .themes()
212            .get(theme_name.as_ref())
213            .cloned()
214        {
215            Theme::global_mut(cx).apply_config(&theme_config);
216            self.settings.app_settings.theme = theme_name;
217            self.settings.app_settings.scrollbar_show = Some(cx.theme().scrollbar_show);
218            let _ = self.update_and_propagate_settings(cx);
219        }
220        cx.refresh_windows();
221        let menus =
222            crate::fulgur::ui::menus::build_menus(self.settings.recent_files.get_files(), None);
223        cx.set_menus(menus);
224    }

226    /// Render the tab bar
227    ///
228    /// ### Arguments
229    /// - `cx`: The application context
230    ///
231    /// ### Returns
232    /// - `Div`: The rendered tab bar element
233    pub fn render_tab_bar(&self, cx: &mut Context<Self>) -> Div {
234        let mut tab_bar = div()
235            .flex()
236            .items_center()
237            .h(TAB_BAR_HEIGHT)
238            .bg(cx.theme().tab_bar);
239        tab_bar = tab_bar
240            .child(
241                tab_bar_button_factory("new-tab", "New Tab", CustomIcon::Plus, cx.theme().border)
242                    .on_click(cx.listener(|this, _, window, cx| {
243                        this.new_tab(window, cx);
244                    })),
245            )
246            .child(
247                tab_bar_button_factory(
248                    "open-file",
249                    "Open File (+Shift - Open Path)",
250                    CustomIcon::FolderOpen,
251                    cx.theme().border,
252                )
253                .on_click(cx.listener(|this, event: &ClickEvent, window, cx| {
254                    if event.modifiers().shift {
255                        this.show_open_from_path_dialog(window, cx);
256                    } else {
257                        this.open_file(window, cx);
258                    }
259                })),
260            )
261            .child(
262                tab_bar_button_factory(
263                    "save-file",
264                    "Save File (+Shift - Save As)",
265                    CustomIcon::Save,
266                    cx.theme().border,
267                )
268                .border_r_1()
269                .on_click(cx.listener(|this, event: &ClickEvent, window, cx| {
270                    if event.modifiers().shift {
271                        this.save_file_as(window, cx);
272                    } else {
273                        this.save_file(window, cx);
274                    }
275                })),
276            )
277            .child(
278                div()
279                    .id("tab-scroll-container")
280                    .overflow_x_scroll()
281                    .track_scroll(&self.tab_scroll_handle)
282                    .flex()
283                    .flex_1()
284                    .items_center()
285                    .children(
286                        self.tabs
287                            .iter()
288                            .enumerate()
289                            .map(|(index, tab)| self.render_tab(index, tab, cx)),
290                    )
291                    .child(
292                        div()
293                            .flex_1()
294                            .min_w(px(0.))
295                            .border_b_1()
296                            .border_color(cx.theme().border)
297                            .h(TAB_BAR_HEIGHT),
298                    ),
299            );
300        tab_bar
301    }

303    /// Render a single tab
304    ///
305    /// ### Arguments
306    /// - `index`: The index of the tab
307    /// - `tab`: The tab to render
308    /// - `cx`: The application context
309    ///
310    /// ### Returns
311    /// - `AnyElement`: The rendered tab element
312    pub fn render_tab(&self, index: usize, tab: &Tab, cx: &mut Context<Self>) -> AnyElement {
313        let tab_id = tab.id();
314        let is_active = match self.active_tab_index {
315            Some(active_index) => index == active_index,
316            None => false,
317        };
318        let has_tabs_on_left = index > 0;
319        let has_tabs_on_right = index < self.tabs.len() - 1;
320        let total_tabs = self.tabs.len();
321        let file_path = tab.as_editor().and_then(|editor_tab| {
322            editor_tab
323                .file_path
324                .as_ref()
325                .and_then(|path| path.to_str().map(|s| s.to_string()))
326        });
327        let cached_file_size = tab
328            .as_editor()
329            .and_then(|editor_tab| editor_tab.file_size_bytes)
330            .map(components_utils::format_file_size);
331        let cached_last_modified = tab
332            .as_editor()
333            .and_then(|editor_tab| editor_tab.file_last_modified)
334            .and_then(components_utils::format_system_time);
335        let mut tab_div = div()
336            .id(("tab", tab_id))
337            .flex()
338            .items_center()
339            .h(TAB_BAR_HEIGHT)
340            .px_2()
341            .gap_2()
342            .border_r_1()
343            .border_b_1()
344            .border_color(cx.theme().border)
345            .on_mouse_down(
346                MouseButton::Left,
347                cx.listener(move |this, _, window, cx: &mut Context<'_, Fulgur>| {
348                    if !is_active {
349                        this.set_active_tab(index, window, cx);
350                    }
351                }),
352            );
353        if is_active {
354            tab_div = tab_div.bg(cx.theme().tab_active).border_b_0();
355            self.set_title(Some(tab.title().to_string()), cx);
356        } else {
357            tab_div = tab_div
358                .bg(cx.theme().tab)
359                .hover(|this| this.bg(cx.theme().muted))
360                .cursor_pointer();
361        }
362        if let Some(path) = file_path {
363            tab_div = tab_div.tooltip(move |window, cx| {
364                let path_clone = path.clone();
365                let file_size = cached_file_size.clone();
366                let last_modified = cached_last_modified.clone();
367                Tooltip::element(move |_, cx| {
368                    let mut tooltip = v_flex().gap_1().py_2().px_1().child(
369                        h_flex()
370                            .gap_3()
371                            .child(CustomIcon::File.icon())
372                            .child(path_clone.clone())
373                            .text_sm()
374                            .font_semibold(),
375                    );
376                    let mut details = h_flex().gap_4().justify_between();
377                    if let Some(ref size) = file_size {
378                        details = details.child(
379                            div()
380                                .child(format!("Size: {}", size))
381                                .text_xs()
382                                .text_color(cx.theme().muted_foreground),
383                        );
384                    }
385                    if let Some(ref last_modified) = last_modified {
386                        details = details.child(
387                            div()
388                                .child(format!("Last Modified: {}", last_modified))
389                                .text_xs()
390                                .text_color(cx.theme().muted_foreground),
391                        );
392                    }
393                    tooltip = tooltip.child(details);
394                    tooltip
395                })
396                .build(window, cx)
397            });
398        }
399        let (filename, folder) = self.get_tab_display_title(index, tab);
400        let modified_indicator = if tab.is_modified() { " â€¢" } else { "" };
401        let mut title_container = div().flex().items_center().gap_1().pl_1().child(
402            div()
403                .text_sm()
404                .text_color(if is_active {
405                    cx.theme().tab_active_foreground
406                } else {
407                    cx.theme().tab_foreground
408                })
409                .child(format!("{}{}", filename, modified_indicator)),
410        );
411        if let Some(folder_path) = folder {
412            title_container = title_container.child(
413                div()
414                    .text_xs()
415                    .italic()
416                    .text_color(if is_active {
417                        cx.theme().tab_active_foreground
418                    } else {
419                        cx.theme().tab_foreground
420                    })
421                    .child(folder_path),
422            );
423        }
424        let tab_with_content = tab_div
425            .child(title_container)
426            .child(
427                Button::new(("close-tab", tab_id))
428                    .icon(CustomIcon::Close)
429                    .ghost()
430                    .xsmall()
431                    .cursor_pointer()
432                    .on_click(cx.listener(move |this, _, window, cx| {
433                        cx.stop_propagation();
434                        this.close_tab(tab_id, window, cx);
435                    })),
436            )
437            .on_mouse_down(
438                MouseButton::Middle,
439                cx.listener(move |this, _, window, cx| {
440                    this.close_tab(tab_id, window, cx);
441                }),
442            )
443            .context_menu(move |this, _window, _cx| {
444                this.menu("Close Tab", Box::new(CloseTabAction(tab_id)))
445                    .menu_with_disabled(
446                        "Close Tabs to the Left",
447                        Box::new(CloseTabsToLeft(index)),
448                        !has_tabs_on_left,
449                    )
450                    .menu_with_disabled(
451                        "Close Tabs to the Right",
452                        Box::new(CloseTabsToRight(index)),
453                        !has_tabs_on_right,
454                    )
455                    .separator()
456                    .menu_with_disabled(
457                        "Close All Tabs",
458                        Box::new(CloseAllTabsAction),
459                        total_tabs == 0,
460                    )
461                    .menu_with_disabled(
462                        "Close All Other Tabs",
463                        Box::new(CloseAllOtherTabs(index)),
464                        total_tabs == 0,
465                    )
466            });
467
468        tab_with_content.into_any_element()
469    }
