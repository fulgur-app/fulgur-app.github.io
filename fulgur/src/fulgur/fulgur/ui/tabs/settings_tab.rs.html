577    /// Create the Themes settings page
578    ///
579    /// ### Arguments
580    /// - `entity`: The Fulgur entity
581    /// - `themes`: The themes to display
582    ///
583    /// ### Returns
584    /// - `SettingPage`: The Themes settings page
585    fn create_themes_page(entity: Entity<Self>, themes: &Themes) -> SettingPage {
586        let mut user_theme_items = Vec::new();
587        let mut default_theme_items = Vec::new();
588        for theme in &themes.user_themes {
589            let theme_name = theme.name.clone();
590            let theme_author = theme.author.clone();
591            let theme_path = theme.path.clone();
592            let themes_info = theme
593                .themes
594                .iter()
595                .map(|t| format!("{} ({})", t.name, t.mode))
596                .collect::<Vec<String>>()
597                .join(", ");
598            let button_id = SharedString::from(format!("delete-theme-{}", theme_name));
599            user_theme_items.push(SettingItem::render({
600                let entity = entity.clone();
601                move |_options, _window, cx| {
602                    let theme_path = theme_path.clone();
603                    let entity_clone = entity.clone();
604                    h_flex()
605                        .w_full()
606                        .justify_between()
607                        .gap_3()
608                        .child(
609                            v_flex()
610                                .gap_1()
611                                .child(
612                                    div()
613                                        .font_semibold()
614                                        .child(format!("{} by {}", theme_name, theme_author)),
615                                )
616                                .child(
617                                    div()
618                                        .text_xs()
619                                        .text_color(cx.theme().muted_foreground)
620                                        .child(themes_info.clone()),
621                                ),
622                        )
623                        .child(
624                            Button::new(button_id.clone())
625                                .icon(CustomIcon::Close)
626                                .small()
627                                .cursor_pointer()
628                                .on_click(move |_, _window, cx| {
629                                    if let Err(e) = fs::remove_file(&theme_path) {
630                                        log::error!(
631                                            "Failed to delete theme file {}: {}",
632                                            theme_path.display(),
633                                            e
634                                        );
635                                    } else {
636                                        log::info!("Deleted theme file: {:?}", theme_path);
637                                    }
638                                    let entity_for_update = entity_clone.clone();
639                                    entity_clone.update(cx, |this, cx| {
640                                        themes::reload_themes_and_update(
641                                            &this.settings,
642                                            entity_for_update,
643                                            cx,
644                                        );
645                                    });
646                                }),
647                        )
648                        .into_any_element()
649                }
650            }));
651        }
652        for theme in &themes.default_themes {
653            let theme_name = theme.name.clone();
654            let theme_author = theme.author.clone();
655            let themes_info = theme
656                .themes
657                .iter()
658                .map(|t| format!("{} ({})", t.name, t.mode))
659                .collect::<Vec<String>>()
660                .join(", ");
661            default_theme_items.push(SettingItem::render(move |_options, _window, cx| {
662                v_flex()
663                    .w_full()
664                    .gap_1()
665                    .child(
666                        div()
667                            .font_semibold()
668                            .child(format!("{} by {} (Default)", theme_name, theme_author)),
669                    )
670                    .child(
671                        div()
672                            .text_xs()
673                            .text_color(cx.theme().muted_foreground)
674                            .child(themes_info.clone()),
675                    )
676                    .into_any_element()
677            }));
678        }
679        let mut groups = Vec::new();
680        if !user_theme_items.is_empty() {
681            groups.push(
682                SettingGroup::new()
683                    .title("User Themes")
684                    .items(user_theme_items),
685            );
686        }
687        if !default_theme_items.is_empty() {
688            groups.push(
689                SettingGroup::new()
690                    .title("Default Themes")
691                    .items(default_theme_items),
692            );
693        }
694        SettingPage::new("Themes").groups(groups)
695    }

697    /// Create settings pages using the Settings component
698    ///
699    /// ### Arguments
700    /// - `window`: The window
701    /// - `cx`: The context
702    ///
703    /// ### Returns
704    /// - `Vec<SettingPage>`: The settings pages
705    fn create_settings_pages(
706        &self,
707        _window: &mut Window,
708        cx: &mut Context<Self>,
709    ) -> Vec<SettingPage> {
710        let entity = cx.entity();
711        let mut pages = vec![
712            Self::create_editor_page(entity.clone()),
713            Self::create_application_page(entity.clone()),
714        ];
715        let themes = self.shared_state(cx).themes.lock().clone();
716        if let Some(ref themes) = themes {
717            pages.push(Self::create_themes_page(entity, themes));
718        }
719        pages
720    }

753    /// Clear the recent files
754    ///
755    /// ### Arguments
756    /// - `cx`: The context
757    pub fn clear_recent_files(&mut self, cx: &mut Context<Self>) {
758        self.settings.recent_files.clear();
759        if let Err(e) = self.update_and_propagate_settings(cx) {
760            log::error!("Failed to save settings: {}", e);
761        }
762        let menus = build_menus(
763            self.settings.recent_files.get_files(),
764            if let Some(info) = self.shared_state(cx).update_info.lock().clone() {
765                Some(info.download_url)
766            } else {
767                None
768            },
769        );
770        cx.set_menus(menus);
771    }
