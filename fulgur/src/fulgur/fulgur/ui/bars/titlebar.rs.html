1// Custom title bar with platform-specific menu bar placement
2use gpui::*;
3#[cfg(not(target_os = "macos"))]
4use gpui_component::menu::AppMenuBar;
5use gpui_component::{ActiveTheme, StyledExt, TitleBar, h_flex};
6
7const DEFAULT_TITLE: &str = "Fulgur";
8
9pub struct CustomTitleBar {
10    #[cfg(not(target_os = "macos"))]
11    app_menu_bar: Entity<AppMenuBar>,
12    title: String,
13}
14
15impl CustomTitleBar {
16    /// Create a new custom title bar
17    ///
18    /// ### Arguments
19    /// - `_window`: The window to create the title bar in
20    /// - `_cx`: The application context
21    ///
22    /// ### Returns
23    /// - `Entity<CustomTitleBar>`: The new custom title bar
24    pub fn new(_window: &mut Window, _cx: &mut App) -> Entity<Self> {
25        #[cfg(not(target_os = "macos"))]
26        let app_menu_bar = AppMenuBar::new(_window, _cx);
27
28        _cx.new(|_cx| Self {
29            #[cfg(not(target_os = "macos"))]
30            app_menu_bar,
31            title: DEFAULT_TITLE.to_string(),
32        })
33    }
34
35    /// Set the title of the title bar
36    ///
37    /// ### Arguments
38    /// - `title`: The optionaltitle to set (if None, the default title is used)
39    pub fn set_title(&mut self, title: Option<String>) {
40        if let Some(title) = title {
41            self.title = format!("{} - Fulgur", title);
42        } else {
43            self.title = DEFAULT_TITLE.to_string();
44        }
45    }
46}
47
48impl Render for CustomTitleBar {
49    /// Render the custom title bar
50    ///
51    /// ### Arguments
52    /// - `_window`: The window to render the title bar in
53    /// - `cx`: The application context
54    ///
55    /// ### Returns
56    /// - `impl IntoElement`: The rendered custom title bar
57    fn render(&mut self, _window: &mut Window, cx: &mut Context<Self>) -> impl IntoElement {
58        let mut title_bar = TitleBar::new().bg(cx.theme().tab_bar);
59        #[cfg(not(target_os = "macos"))]
60        {
61            title_bar =
62                title_bar.child(div().flex().items_center().child(self.app_menu_bar.clone()));
63        }
64        title_bar = title_bar.child(
65            h_flex().flex_1().justify_center().items_center().child(
66                div()
67                    .text_sm()
68                    .font_semibold()
69                    .text_color(cx.theme().foreground)
70                    .child(self.title.clone()),
71            ),
72        );
73        #[cfg(not(target_os = "macos"))]
74        {
75            title_bar = title_bar.child(div().w_40());
76        }
77        #[cfg(target_os = "macos")]
78        {
79            title_bar = title_bar.child(div().w_20());
80        }
81        title_bar
82    }
83}
