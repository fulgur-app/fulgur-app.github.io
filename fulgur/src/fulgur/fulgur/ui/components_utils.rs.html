42/// Create a button
43///
44/// ### Arguments
45/// - `id`: The ID of the button
46/// - `tooltip`: The tooltip of the button for the button
47/// - `icon`: The icon of the button
48/// - `border_color`: The color of the border of the button
49///
50/// ### Returns
51/// - `Button`: The button
52pub fn button_factory(
53    id: &'static str,
54    tooltip: &'static str,
55    icon: CustomIcon,
56    border_color: Hsla,
57) -> Button {
58    Button::new(id)
59        .icon(icon)
60        .text()
61        .small()
62        .tooltip(tooltip)
63        .ghost()
64        .p_0()
65        .m_0()
66        .border_0()
67        .border_color(border_color)
68        .cursor_pointer()
69        .corner_radii(CORNERS_SIZE)
70}

72/// Format a date as ISO 8601 string
73///
74/// ### Arguments
75/// - `time`: The time to format
76///
77/// ### Returns
78/// - `Some(String)`: The formatted date
79/// - `None`: If the time could not be formatted
80pub fn format_system_time(time: SystemTime) -> Option<String> {
81    let datetime = OffsetDateTime::from(time);
82    let format =
83        time::format_description::parse("[year]-[month]-[day] [hour]:[minute]:[second]").ok()?;
84    datetime.format(&format).ok()
85}

87/// Format file size in a human-readable format.
88///
89/// ### Arguments
90/// - `bytes`: File size in bytes
91///
92/// ### Returns
93/// - `String`: Human-readable file size
94pub fn format_file_size(bytes: u64) -> String {
95    if bytes < 1024 {
96        format!("{} B", bytes)
97    } else if bytes < 1024 * 1024 {
98        format!("{:.1} KB", bytes as f64 / 1024.0)
99    } else {
100        format!("{:.1} MB", bytes as f64 / (1024.0 * 1024.0))
101    }
102}
