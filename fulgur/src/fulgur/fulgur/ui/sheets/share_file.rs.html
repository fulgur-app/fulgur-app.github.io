22/// Create a device item for the share file sheet
23///
24/// ### Arguments
25/// - `device`: The device to display
26/// - `is_selected`: Whether the device is selected
27/// - `selected_ids`: The shared state of selected device IDs
28/// - `idx`: The index of the device (for unique ID)
29/// - `cx`: The application context
30///
31/// ### Returns
32/// - `impl Element`: The device item element (a Div, actually)
33fn make_device_item(
34    device: &Device,
35    is_selected: bool,
36    selected_ids: Arc<parking_lot::Mutex<Vec<String>>>,
37    idx: usize,
38    cx: &App,
39) -> impl Element {
40    let device_id = device.id.clone();
41    let device_name = device.name.clone();
42    let device_expires = device.expires_at.clone();
43    let device_for_icon = device;
44    let has_public_key = device.public_key.is_some();
45    h_flex()
46        .id(("share-device-sheet", idx))
47        .items_center()
48        .justify_between()
49        .w_full()
50        .p_2()
51        .my_2()
52        .rounded_sm()
53        .border_color(cx.theme().border)
54        .border_1()
55        .when(has_public_key, |this| this.cursor_pointer())
56        .when(!has_public_key, |this| this.opacity(0.5))
57        .when(has_public_key, |this| {
58            this.hover(|hover| hover.bg(cx.theme().muted))
59        })
60        .when(is_selected && has_public_key, |this| {
61            this.bg(cx.theme().accent)
62                .text_color(cx.theme().accent_foreground)
63        })
64        .child(
65            v_flex()
66                .gap_1()
67                .child(
68                    h_flex()
69                        .items_center()
70                        .justify_start()
71                        .gap_2()
72                        .child(get_icon(device_for_icon))
73                        .child(div().child(device_name))
74                        .child(
75                            div()
76                                .text_xs()
77                                .child(format!("Expires: {}", device_expires)),
78                        ),
79                )
80                .when(!has_public_key, |this| {
81                    this.child(
82                        div()
83                            .text_xs()
84                            .text_color(cx.theme().muted_foreground)
85                            .child("No public key for this device"),
86                    )
87                }),
88        )
89        .when(is_selected && has_public_key, |this| {
90            this.child(Icon::new(CustomIcon::Zap))
91        })
92        .when(has_public_key, |this| {
93            this.on_click(move |_event, _window, _cx| {
94                let mut ids = selected_ids.lock();
95                if let Some(pos) = ids.iter().position(|id| id == &device_id) {
96                    ids.remove(pos);
97                } else {
98                    ids.push(device_id.clone());
99                }
100            })
101        })
102}

104/// Create the device list for the share file sheet
105///
106/// ### Arguments
107/// - `devices`: The list of devices (wrapped in Arc)
108/// - `selected_ids`: The shared state of selected device IDs
109/// - `cx`: The application context
110///
111/// ### Returns
112/// - `Div`: The list of devices
113fn make_device_list(
114    devices: Arc<Vec<Device>>,
115    selected_ids: Arc<parking_lot::Mutex<Vec<String>>>,
116    cx: &App,
117) -> Div {
118    div().gap_2().children(
119        devices
120            .iter()
121            .enumerate()
122            .map(|(idx, device)| {
123                let is_selected = selected_ids.lock().contains(&device.id);
124                make_device_item(device, is_selected, selected_ids.clone(), idx, cx)
125            })
126            .collect::<Vec<_>>(),
127    )
128}

130/// Handle the share file action when OK is clicked
131///
132/// ### Arguments
133/// - `selected_ids`: The selected device IDs
134/// - `devices`: The list of all devices (with their public keys)
135/// - `entity`: The Fulgur entity
136/// - `window`: The window context
137/// - `cx`: The application context
138fn handle_share_file(
139    selected_ids: Arc<parking_lot::Mutex<Vec<String>>>,
140    devices: Arc<Vec<Device>>,
141    entity: Entity<Fulgur>,
142    window: &mut Window,
143    cx: &mut App,
144) {
145    let ids = selected_ids.lock().clone();
146    if ids.is_empty() {
147        window.push_notification(
148            (
149                NotificationType::Warning,
150                SharedString::from("Please select at least one device to share with."),
151            ),
152            cx,
153        );
154        return;
155    }
156    let result = entity.update(cx, |this, cx| {
157        let active_tab = this.get_active_editor_tab();
158        let content = active_tab
159            .as_ref()
160            .map(|tab| tab.content.read(cx).value().to_string())
161            .unwrap_or_default();
162        let file_path = active_tab.as_ref().and_then(|tab| tab.file_path.clone());
163        let file_name = file_path
164            .as_ref()
165            .and_then(|path| path.file_name())
166            .and_then(|name| name.to_str())
167            .unwrap_or("Untitled")
168            .to_string();
169        share_file(
170            &this.settings.app_settings.synchronization_settings,
171            ShareFileRequest {
172                content,
173                file_name,
174                device_ids: ids,
175                file_path,
176            },
177            &devices,
178            Arc::clone(&this.shared_state(cx).sync_state.token_state),
179            &this.shared_state(cx).http_agent,
180        )
181    });
182    match result {
183        Ok(share_result) => {
184            if share_result.is_complete_success() {
185                let notification = (
186                    NotificationType::Success,
187                    SharedString::from(share_result.summary_message()),
188                );
189                window.push_notification(notification, cx);
190                window.close_sheet(cx);
191            } else if share_result.successes.is_empty() {
192                let notification = (
193                    NotificationType::Error,
194                    SharedString::from(share_result.summary_message()),
195                );
196                window.push_notification(notification, cx);
197            } else {
198                let notification = (
199                    NotificationType::Warning,
200                    SharedString::from(share_result.summary_message()),
201                );
202                window.push_notification(notification, cx);
203                window.close_sheet(cx);
204            }
205        }
206        Err(e) => {
207            log::error!("Failed to share file: {}", e);
208            let notification = (
209                NotificationType::Error,
210                SharedString::from(format!("Failed to share file: {}", e)),
211            );
212            window.push_notification(notification, cx);
213        }
214    }
215}

217impl Fulgur {
218    /// Open the share file sheet
219    ///
220    /// ### Arguments
221    /// - `window`: The window context
222    /// - `cx`: The application context
223    pub fn open_share_file_sheet(&mut self, window: &mut Window, cx: &mut Context<Self>) {
224        if !self
225            .settings
226            .app_settings
227            .synchronization_settings
228            .is_synchronization_activated
229        {
230            log::warn!("Synchronization is not activated");
231            return;
232        }
233        if !self.is_connected(cx) {
234            log::info!("Not connected to sync server, attempting reconnection...");
235            window.push_notification(
236                (
237                    NotificationType::Info,
238                    SharedString::from("Attempting to connect to sync server..."),
239                ),
240                cx,
241            );
242            let synchronization_settings =
243                self.settings.app_settings.synchronization_settings.clone();
244            let token_state = Arc::clone(&self.shared_state(cx).sync_state.token_state);
245            let http_agent = &self.shared_state(cx).http_agent;
246            let result = crate::fulgur::sync::synchronization::initial_synchronization(
247                &synchronization_settings,
248                token_state,
249                http_agent,
250            );
251            match result {
252                Ok(begin_response) => {
253                    {
254                        let mut device_name = self.shared_state(cx).sync_state.device_name.lock();
255                        *device_name = Some(begin_response.device_name.clone());
256                    }
257                    {
258                        let mut files =
259                            self.shared_state(cx).sync_state.pending_shared_files.lock();
260                        *files = begin_response.shares;
261                    }
262                    *self.shared_state(cx).sync_state.connection_status.lock() =
263                        SynchronizationStatus::Connected;
264                    self.restart_sse_connection(cx);
265                }
266                Err(e) => {
267                    let connection_status = SynchronizationStatus::from_error(&e);
268                    *self.shared_state(cx).sync_state.connection_status.lock() = connection_status;
269                    let dialog_message = match connection_status {
270                        SynchronizationStatus::AuthenticationFailed => {
271                            "Authentication failed. Check your e-mail and device API key in the synchronization settings."
272                        }
273                        SynchronizationStatus::ConnectionFailed => {
274                            "Connection failed. Check the server URL in the synchronization settings."
275                        }
276                        SynchronizationStatus::Other => {
277                            "An error occurred while connecting to the sync server. Check your synchronization settings."
278                        }
279                        SynchronizationStatus::NotActivated => {
280                            "Synchronization is not activated. You can activate synchronization in the settings."
281                        }
282                        SynchronizationStatus::Disconnected => {
283                            "Could not connect to the sync server. Check your synchronization settings."
284                        }
285                        SynchronizationStatus::Connected => "Unknown error occurred.",
286                    };
287                    window
288                        .open_dialog(cx, move |dialog, _, _| dialog.alert().child(dialog_message));
289                    return;
290                }
291            }
292        }
293        let synchronization_settings = self.settings.app_settings.synchronization_settings.clone();
294        let devices = get_devices(
295            &synchronization_settings,
296            Arc::clone(&self.shared_state(cx).sync_state.token_state),
297            &self.shared_state(cx).http_agent,
298        );
299        let devices = match devices {
300            Ok(devices) => devices,
301            Err(e) => {
302                log::error!("Failed to get devices: {}", e);
303                window.push_notification(
304                    (
305                        NotificationType::Error,
306                        SharedString::from(format!("Failed to get devices: {}", e)),
307                    ),
308                    cx,
309                );
310                return;
311            }
312        };
313        let entity = cx.entity();
314        let selected_ids: Arc<parking_lot::Mutex<Vec<String>>> =
315            Arc::new(parking_lot::Mutex::new(Vec::new()));
316        let devices: Arc<Vec<Device>> = Arc::new(devices);
317        let viewport_height = window.viewport_size().height;
318        window.open_sheet(cx, move |sheet, _window, cx| {
319            let max_height = px((viewport_height - px(150.0)).into());
320            let selected_ids_for_list = selected_ids.clone();
321            let selected_ids_for_ok = selected_ids.clone();
322            let entity_for_ok = entity.clone();
323            let devices_ref = devices.clone();
324            let devices_button = devices.clone();
325            sheet
326                .title("Share with...")
327                .size(px(400.))
328                .overlay(true)
329                .child(
330                    v_flex()
331                        .overflow_y_scrollbar()
332                        .gap_2()
333                        .h(max_height)
334                        .child(make_device_list(devices_ref, selected_ids_for_list, cx)),
335                )
336                .footer(
337                    h_flex()
338                        .justify_end()
339                        .w_full()
340                        .gap_2()
341                        .child(
342                            Button::new("cancel-share")
343                                .child("Cancel")
344                                .small()
345                                .cursor_pointer()
346                                .on_click(|_, window, cx| {
347                                    window.close_sheet(cx);
348                                }),
349                        )
350                        .child(
351                            Button::new("ok-share")
352                                .child("Share")
353                                .small()
354                                .primary()
355                                .cursor_pointer()
356                                .on_click(move |_, window, cx| {
357                                    handle_share_file(
358                                        selected_ids_for_ok.clone(),
359                                        devices_button.clone(),
360                                        entity_for_ok.clone(),
361                                        window,
362                                        cx,
363                                    );
364                                }),
365                        ),
366                )
367        });
368    }
