22async fn main() -> anyhow::Result<()> {
23    dotenv().ok();
24    let is_prod = std::env::var("IS_PROD").unwrap_or("false".to_string()) == "true";
25    let log_folder = logging::get_log_folder();
26    let _log_guard = logging::init_logging(log_folder.clone(), is_prod)?;

28    tracing::info!("Starting Fulgur server v{}", env!("CARGO_PKG_VERSION"));
29    tracing::info!(
30        "Running in {} environment",
31        if is_prod { "PRODUCTION" } else { "DEVELOPMENT" }
32    );
33    tracing::info!("Log folder: {}", log_folder.display());
34    tracing::info!("========================================");
35    let database_url = std::env::var("DATABASE_URL").expect("DATABASE_URL must be set");

37        .create_if_missing(true)
38        .journal_mode(sqlx::sqlite::SqliteJournalMode::Wal)
39        .busy_timeout(std::time::Duration::from_secs(30));
40    let connection = SqlitePoolOptions::new()
41        .max_connections(5)
42        .connect_with(options)
43        .await?;
44    tracing::info!("Database connection established");
45    tracing::info!("Running migrations...");
46    sqlx::migrate!("./data/migrations").run(&connection).await?;
47    tracing::info!("Migrations completed successfully");
48    let device_repository = devices::DeviceRepository::new(connection.clone());
49    let user_repository = users::UserRepository::new(connection.clone());
50    let verification_code_repository =
51        verification_code::VerificationCodeRepository::new(connection.clone());
52    let share_repository = shares::ShareRepository::new(connection.clone());
53    let has_admin = user_repository.has_admin().await?;
54    let setup_needed = !has_admin;
55    if setup_needed {
56        tracing::warn!("No admin user found - initial setup required at /setup");
57    }

59        .unwrap_or_else(|_| "30".to_string())
60        .parse::<u64>()

62    tracing::info!("SSE heartbeat interval: {} seconds", sse_heartbeat_seconds);
63    let sse_manager = Arc::new(api::sse::SseChannelManager::new());
64    tracing::info!("SSE channel manager initialized");
65    let jwt_secret = std::env::var("JWT_SECRET")
66        .expect("JWT_SECRET must be set. Generate with: openssl rand -base64 32");
67    if jwt_secret.len() < 32 {
68        panic!("JWT_SECRET must be at least 32 characters for security");
69    }
70    tracing::info!(
71        "JWT secret loaded (length: {} characters)",
72        jwt_secret.len()
73    );
74    let jwt_expiry_seconds = std::env::var("JWT_EXPIRY_SECONDS")
75        .unwrap_or_else(|_| "900".to_string())
76        .parse::<i64>()
77        .expect("JWT_EXPIRY_SECONDS must be a valid number");
78    tracing::info!(
79        "JWT access token expiry: {} seconds (~{} minutes)",
80        jwt_expiry_seconds,
81        jwt_expiry_seconds / 60
82    );
83    let app_state = handlers::AppState {
84        device_repository,
85        user_repository,
86        verification_code_repository,
87        share_repository,
88        mailer: Arc::new(mail::Mailer::new(is_prod)),
89        is_prod,
90        can_register: auth::handlers::can_register(),
91        setup_needed: Arc::new(AtomicBool::new(setup_needed)),
92        share_validity_days: shares::get_share_validity_days(),
93        max_devices_per_user: devices::get_max_devices_per_user(),
94        sse_manager,
95        sse_heartbeat_seconds,
96        jwt_secret,
97        jwt_expiry_seconds,

99    tracing::info!("Max devices per user: {}", app_state.max_devices_per_user);
100    tracing::info!("API rate limiter: 100 requests per minute per IP");
101    tracing::info!("Auth rate limiter: 10 requests per minute per IP");

103    let tls_key_path = std::env::var("TLS_KEY_PATH").ok();

107        .with_secure(tls_enabled)
108        .with_expiry(Expiry::OnInactivity(CookieDuration::hours(1)));
109    if tls_enabled {
110        tracing::info!("Session cookies configured with secure flag (HTTPS only)");
111    } else {
112        tracing::warn!("Session cookies configured without secure flag (HTTP mode)");
113    }
114    let app = fulgurant::build_app(&app_state, session_layer);
115    let assets_service = ServeDir::new("assets");
116    let app = app.nest_service("/assets", assets_service);

118    let cleanup_share_repo = app_state.share_repository.clone();

120    let cleanup_verification_repo = app_state.verification_code_repository.clone();

127    let bind_port = std::env::var("BIND_PORT")
128        .unwrap_or_else(|_| "3000".to_string())
129        .parse::<u16>()
130        .expect("BIND_PORT must be a valid port number");
131    let addr = format!("{}:{}", bind_host, bind_port)
132        .parse::<SocketAddr>()
133        .expect("Failed to parse bind address");
134    if bind_host == "0.0.0.0" {
135        tracing::warn!("Server is listening on all interfaces (0.0.0.0)");
136    } else if bind_host == "127.0.0.1" {
137        tracing::info!("Server is listening on localhost only");
138    }
139    let sse_manager_shutdown = app_state.sse_manager.clone();
140    let shutdown_token_handler = shutdown_token.clone();
141    tokio::spawn(async move {
142        shutdown_signal().await;
143        shutdown_token_handler.cancel();
144        tracing::info!("Closing all SSE connections");
145        drop(sse_manager_shutdown);
146        tracing::info!("Graceful shutdown complete");
147    });
148
149    match (tls_cert_path, tls_key_path) {
150        (Some(cert_path), Some(key_path)) => {
151            tracing::info!("TLS enabled - loading certificate and key");
152            tracing::info!("Certificate: {}", cert_path);
153            tracing::info!("Private key: {}", key_path);
154            let _ = rustls::crypto::aws_lc_rs::default_provider().install_default();
155            let tls_config =
156                axum_server::tls_rustls::RustlsConfig::from_pem_file(cert_path, key_path)
157                    .await
158                    .expect("Failed to load TLS certificate and key");
159            tracing::info!("Server starting on https://{}", addr);
160            axum_server::bind_rustls(addr, tls_config)
161                .serve(app.into_make_service_with_connect_info::<SocketAddr>())
162                .await?;
163        }
164        _ => {
165            tracing::info!("TLS disabled - running HTTP only");
166            tracing::warn!(
167                "For production, configure TLS_CERT_PATH and TLS_KEY_PATH environment variables"
168            );
169            tracing::info!("Server starting on http://{}", addr);
170            let listener = tokio::net::TcpListener::bind(addr).await?;
171            axum::serve(
172                listener,
173                app.into_make_service_with_connect_info::<SocketAddr>(),
174            )
175            .with_graceful_shutdown(shutdown_signal())
176            .await?;
177        }
178    }
179    Ok(())
180}

182/// Create a shutdown signal handler that listens for SIGTERM/SIGINT
183async fn shutdown_signal() {
184    let ctrl_c = async {
185        tokio::signal::ctrl_c()
186            .await
187            .expect("Failed to install Ctrl+C handler");
188    };
189
190    #[cfg(unix)]
191    let terminate = async {
192        tokio::signal::unix::signal(tokio::signal::unix::SignalKind::terminate())
193            .expect("Failed to install SIGTERM handler")
194            .recv()
195            .await;
196    };
197
198    #[cfg(not(unix))]
199    let terminate = std::future::pending::<()>();
200
201    tokio::select! {
202        _ = ctrl_c => {
203            tracing::info!("Received Ctrl+C signal");
204        },
205        _ = terminate => {
206            tracing::info!("Received SIGTERM signal");
207        },
208    }
209
210    tracing::info!("Starting graceful shutdown...");
211}

213/// Make the share cleanup task. Runs every hour.
214///
215/// ### Arguments
216/// - `share_repository`: The share repository
217/// - `shutdown_token`: Token to signal graceful shutdown
218fn make_share_cleanup_task(share_repository: ShareRepository, shutdown_token: CancellationToken) {
219    tracing::info!("Starting share cleanup task (runs every 1 hour)");
220    tokio::spawn(async move {
221        let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(3600)); // 1 hour
222        loop {
223            tokio::select! {
224                _ = interval.tick() => {
225                    match share_repository.delete_expired().await {
226                        Ok(count) => {
227                            if count > 0 {
228                                tracing::info!("Cleaned up {} expired share(s)", count);
229                            } else {
230                                tracing::debug!("Share cleanup check complete - no expired shares found");
231                            }
232                        }
233                        Err(e) => {
234                            tracing::error!("Error cleaning up expired shares: {:?}", e);
235                        }
236                    }
237                },
238                _ = shutdown_token.cancelled() => {
239                    tracing::info!("Share cleanup task shutting down gracefully");
240                    break;
241                }
242            }
243        }
244    });
245}
