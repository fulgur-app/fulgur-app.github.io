47    /// Convert the share to a display share
48    ///
49    /// ### Arguments
50    /// - `devices`: The devices to convert the share to a display share
51    ///
52    /// ### Returns
53    /// - `DisplayShare`: The display share
54    pub fn to_display_shares(&self, devices: &[Device]) -> DisplayShare {
55        let from = match devices
56            .iter()
57            .find(|d| d.device_id == self.source_device_id)
58        {
59            Some(d) => d.name.clone(),
60            None => "Unknown".to_string(),
61        };
62        let to = match devices
63            .iter()
64            .find(|d| d.device_id == self.destination_device_id)
65        {
66            Some(d) => d.name.clone(),
67            None => "Unknown".to_string(),
68        };
69        let created_at = format_datetime_utc(&self.created_at);
70        let expires_at = format_datetime_utc(&self.expires_at);
71        DisplayShare {
72            id: self.id.clone(),
73            file_name: self.file_name.clone(),
74            from,
75            to,
76            created_at,
77            expires_at,
78        }
79    }

100/// Calculate SHA256 hash of file content
101///
102/// ### Arguments
103/// - `content`: The content to calculate the hash of a file
104///
105/// ### Returns
106/// - `String`: The SHA256 hash of the content
107pub fn calculate_file_hash(content: &str) -> String {
108    let mut hasher = Sha256::new();
109    hasher.update(content.as_bytes());
110    format!("{:x}", hasher.finalize())
111}

118impl ShareRepository {
119    /// Create a new share repository
120    ///
121    /// ### Arguments
122    /// - `pool`: The SQLite pool
123    ///
124    /// ### Returns
125    /// - `ShareRepository`: The share repository
126    pub fn new(pool: Pool<Sqlite>) -> Self {
127        Self { pool }
128    }
129
130    /// Create a new share, or replace an existing one if deduplication_hash matches
131    ///
132    /// When `deduplication_hash` is `Some`, uses UPSERT to replace any existing share
133    /// with the same (source_device_id, destination_device_id, deduplication_hash).
134    /// When `None`, always inserts a new share (SQLite treats NULLs as distinct).
135    ///
136    /// ### Arguments
137    /// - `user_id`: The ID of the user
138    /// - `data`: The data for the share
139    ///
140    /// ### Returns
141    /// - `Ok(Share)`: The created or updated share
142    /// - `Err(sqlx::Error)`: The error that occurred while creating the share
143    pub async fn create(&self, user_id: i32, data: CreateShare) -> Result<Share, sqlx::Error> {
144        let id = Uuid::new_v4().to_string();
145        let file_hash = calculate_file_hash(&data.content);
146        let file_size = data.content.len() as i32;
147        let now = OffsetDateTime::now_utc();
148        let expires_at = now + Duration::days(SHARE_VALIDITY_DAYS);
149
150        if data.deduplication_hash.is_some() {
151            // UPSERT: replace existing share with same source/destination/hash
152            let share = sqlx::query_as::<_, Share>(
153                r#"
154                INSERT INTO shares (
155                    id, user_id, source_device_id, destination_device_id,
156                    file_hash, file_name, file_size, content, deduplication_hash,
157                    created_at, expires_at
158                ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
159                ON CONFLICT(source_device_id, destination_device_id, deduplication_hash)
160                DO UPDATE SET
161                    file_hash = excluded.file_hash,
162                    file_name = excluded.file_name,
163                    file_size = excluded.file_size,
164                    content = excluded.content,
165                    created_at = excluded.created_at,
166                    expires_at = excluded.expires_at
167                RETURNING *
168                "#,
169            )
170            .bind(&id)
171            .bind(user_id)
172            .bind(&data.source_device_id)
173            .bind(&data.destination_device_id)
174            .bind(&file_hash)
175            .bind(&data.file_name)
176            .bind(file_size)
177            .bind(&data.content)
178            .bind(&data.deduplication_hash)
179            .bind(now.unix_timestamp())
180            .bind(expires_at.unix_timestamp())
181            .fetch_one(&self.pool)
182            .await?;
183            Ok(share)
184        } else {
185            // Plain INSERT: no dedup, always creates a new share
186            let share = sqlx::query_as::<_, Share>(
187                r#"
188                INSERT INTO shares (
189                    id, user_id, source_device_id, destination_device_id,
190                    file_hash, file_name, file_size, content, deduplication_hash,
191                    created_at, expires_at
192                ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, NULL, ?, ?)
193                RETURNING *
194                "#,
195            )
196            .bind(&id)
197            .bind(user_id)
198            .bind(&data.source_device_id)
199            .bind(&data.destination_device_id)
200            .bind(&file_hash)
201            .bind(&data.file_name)
202            .bind(file_size)
203            .bind(&data.content)
204            .bind(now.unix_timestamp())
205            .bind(expires_at.unix_timestamp())
206            .fetch_one(&self.pool)
207            .await?;
208            Ok(share)
209        }
210    }
