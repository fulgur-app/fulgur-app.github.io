44/// Build the complete application router
45///
46/// ### Arguments
47/// - `app_state`: The application state containing repositories and configuration
48/// - `session_layer`: The session management layer
49///
50/// ### Returns
51/// - `Router`: The fully assembled router with all routes and middleware
52pub fn build_app(
53    app_state: &handlers::AppState,
54    session_layer: SessionManagerLayer<MemoryStore>,
55) -> Router {
56    let auth_routes = make_auth_routes(app_state, session_layer.clone());
57    let api_routes = make_api_routes(app_state);
58    let web_routes = make_web_routes(app_state, session_layer);
59
60    Router::new()
61        .merge(auth_routes)
62        .merge(web_routes)
63        .merge(api_routes)
64        .fallback(handlers::not_found)
65}

67/// Make the auth routes
68///
69/// ### Arguments
70/// - `app_state`: The state of the application
71/// - `session_layer`: The session layer
72///
73/// ### Returns
74/// - `Router`: The router that handles the auth routes
75fn make_auth_routes(
76    app_state: &handlers::AppState,
77    session_layer: SessionManagerLayer<MemoryStore>,
78) -> Router {
79    let auth_governor_conf = Arc::new(
80        tower_governor::governor::GovernorConfigBuilder::default()
81            .period(std::time::Duration::from_secs(6)) // 10 requests/min = 1 per 6s
82            .burst_size(5)
83            .use_headers()
84            .finish()
85            .expect("Failed to build auth governor config"),
86    );
87    Router::new()
88        .route("/auth/register", post(auth::handlers::register_step_1))
89        .route(
90            "/auth/register/step2",
91            post(auth::handlers::register_step_2),
92        )
93        .route("/login", post(auth::handlers::login))
94        .route(
95            "/auth/forgot-password",
96            post(auth::handlers::forgot_password_step_1),
97        )
98        .route(
99            "/auth/forgot-password/verify",
100            post(auth::handlers::forgot_password_step_2),
101        )
102        .route(
103            "/auth/forgot-password/reset",
104            post(auth::handlers::forgot_password_step_3),
105        )
106        .route(
107            "/auth/forgot-password/resend",
108            get(auth::handlers::resend_forgot_password_code),
109        )
110        .route(
111            "/force-password-update",
112            post(auth::handlers::force_password_update),
113        )
114        .route("/setup", post(setup::handlers::create_admin))
115        .with_state(app_state.clone())
116        .layer(axum::middleware::from_fn_with_state(
117            app_state.clone(),
118            setup::middleware::require_setup_complete,
119        ))
120        .layer(axum::middleware::from_fn(
121            axum_tower_sessions_csrf::CsrfMiddleware::middleware,
122        ))
123        .layer(session_layer)
124        .layer(tower_governor::GovernorLayer::new(auth_governor_conf))
125}

184/// Make the api routes
185///
186/// ### Arguments
187/// - `app_state`: The state of the application
188///
189/// ### Returns
190/// - `Router`: The router that handles the api routes
191fn make_api_routes(app_state: &handlers::AppState) -> Router {
192    let governor_conf = Arc::new(
193        tower_governor::governor::GovernorConfigBuilder::default()
194            .period(std::time::Duration::from_millis(600)) // 100 requests/min = 1 per 600ms
195            .burst_size(20)
196            .use_headers()
197            .finish()
198            .expect("Failed to build governor config"),
199    );
200    let token_route = Router::new()
201        .route("/api/token", post(api::handlers::obtain_access_token))
202        .layer(tower_governor::GovernorLayer::new(governor_conf.clone()))
203        .with_state(app_state.clone());
204    let authenticated_routes = Router::new()
205        .route("/api/ping", get(api::handlers::ping))
206        .route("/api/begin", post(api::handlers::begin))
207        .route("/api/devices", get(api::handlers::get_devices))
208        .route("/api/share", post(api::handlers::share_file))
209        .route("/api/shares", get(api::handlers::get_shares))
210        .route("/api/sse", get(api::sse::handle_sse_connection))
211        .layer(axum_mw::from_fn_with_state(
212            app_state.clone(),
213            api::middleware::require_api_auth,
214        ))
215        .layer(tower_governor::GovernorLayer::new(governor_conf))
216        .with_state(app_state.clone());
217    token_route.merge(authenticated_routes)
218}

220/// Make the web routes
221///
222/// ### Arguments
223/// - `app_state`: The state of the application
224/// - `session_layer`: The session layer
225///
226/// ### Returns
227/// - `Router`: The router that handles the web routes
228fn make_web_routes(
229    app_state: &handlers::AppState,
230    session_layer: SessionManagerLayer<MemoryStore>,
231) -> Router {
232    Router::new()
233        .merge(make_public_routes(app_state))
234        .merge(make_protected_routes(app_state))
235        .layer(axum::middleware::from_fn(
236            axum_tower_sessions_csrf::CsrfMiddleware::middleware,
237        ))
238        .layer(session_layer)
239        .layer(SetResponseHeaderLayer::if_not_present(
240            header::X_FRAME_OPTIONS,
241            HeaderValue::from_static("DENY"),
242        ))
243        .layer(SetResponseHeaderLayer::if_not_present(
244            header::X_CONTENT_TYPE_OPTIONS,
245            HeaderValue::from_static("nosniff"),
246        ))
247        .layer(SetResponseHeaderLayer::if_not_present(
248            header::HeaderName::from_static("x-xss-protection"),
249            HeaderValue::from_static("1; mode=block"),
250        ))
251        .layer(SetResponseHeaderLayer::if_not_present(
252            header::STRICT_TRANSPORT_SECURITY,
253            HeaderValue::from_static("max-age=31536000; includeSubDomains"),
254        ))
255        .layer(axum::middleware::from_fn(csp::add_csp_nonce_and_header))
256}
