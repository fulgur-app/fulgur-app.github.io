43/// Build the complete application router
44///
45/// ### Arguments
46/// - `app_state`: The application state containing repositories and configuration
47/// - `session_layer`: The session management layer
48///
49/// ### Returns
50/// - `Router`: The fully assembled router with all routes and middleware
51pub fn build_app(
52    app_state: &handlers::AppState,
53    session_layer: SessionManagerLayer<MemoryStore>,
54) -> Router {
55    let auth_routes = make_auth_routes(app_state, session_layer.clone());
56    let api_routes = make_api_routes(app_state);
57    let web_routes = make_web_routes(app_state, session_layer);
58
59    Router::new()
60        .merge(auth_routes)
61        .merge(web_routes)
62        .merge(api_routes)
63        .fallback(handlers::not_found)
64}

66/// Make the auth routes
67///
68/// ### Arguments
69/// - `app_state`: The state of the application
70/// - `session_layer`: The session layer
71///
72/// ### Returns
73/// - `Router`: The router that handles the auth routes
74fn make_auth_routes(
75    app_state: &handlers::AppState,
76    session_layer: SessionManagerLayer<MemoryStore>,
77) -> Router {
78    let auth_governor_conf = Arc::new(
79        tower_governor::governor::GovernorConfigBuilder::default()
80            .period(std::time::Duration::from_secs(6)) // 10 requests/min = 1 per 6s
81            .burst_size(5)
82            .use_headers()
83            .finish()
84            .expect("Failed to build auth governor config"),
85    );
86    Router::new()
87        .route("/auth/register", post(auth::handlers::register_step_1))
88        .route(
89            "/auth/register/step2",
90            post(auth::handlers::register_step_2),
91        )
92        .route("/login", post(auth::handlers::login))
93        .route(
94            "/auth/forgot-password",
95            post(auth::handlers::forgot_password_step_1),
96        )
97        .route(
98            "/auth/forgot-password/verify",
99            post(auth::handlers::forgot_password_step_2),
100        )
101        .route(
102            "/auth/forgot-password/reset",
103            post(auth::handlers::forgot_password_step_3),
104        )
105        .route(
106            "/auth/forgot-password/resend",
107            get(auth::handlers::resend_forgot_password_code),
108        )
109        .route(
110            "/force-password-update",
111            post(auth::handlers::force_password_update),
112        )
113        .route("/setup", post(setup::handlers::create_admin))
114        .with_state(app_state.clone())
115        .layer(axum::middleware::from_fn_with_state(
116            app_state.clone(),
117            setup::middleware::require_setup_complete,
118        ))
119        .layer(axum::middleware::from_fn(
120            axum_tower_sessions_csrf::CsrfMiddleware::middleware,
121        ))
122        .layer(session_layer)
123        .layer(tower_governor::GovernorLayer::new(auth_governor_conf))
124}

180/// Make the api routes
181///
182/// ### Arguments
183/// - `app_state`: The state of the application
184///
185/// ### Returns
186/// - `Router`: The router that handles the api routes
187fn make_api_routes(app_state: &handlers::AppState) -> Router {
188    let governor_conf = Arc::new(
189        tower_governor::governor::GovernorConfigBuilder::default()
190            .period(std::time::Duration::from_millis(600)) // 100 requests/min = 1 per 600ms
191            .burst_size(20)
192            .use_headers()
193            .finish()
194            .expect("Failed to build governor config"),
195    );
196    let token_route = Router::new()
197        .route("/api/token", post(api::handlers::obtain_access_token))
198        .layer(tower_governor::GovernorLayer::new(governor_conf.clone()))
199        .with_state(app_state.clone());
200    let authenticated_routes = Router::new()
201        .route("/api/ping", get(api::handlers::ping))
202        .route("/api/begin", post(api::handlers::begin))
203        .route("/api/devices", get(api::handlers::get_devices))
204        .route("/api/share", post(api::handlers::share_file))
205        .route("/api/shares", get(api::handlers::get_shares))
206        .route("/api/sse", get(api::sse::handle_sse_connection))
207        .layer(axum_mw::from_fn_with_state(
208            app_state.clone(),
209            api::middleware::require_api_auth,
210        ))
211        .layer(tower_governor::GovernorLayer::new(governor_conf))
212        .with_state(app_state.clone());
213    token_route.merge(authenticated_routes)
214}

216/// Make the web routes
217///
218/// ### Arguments
219/// - `app_state`: The state of the application
220/// - `session_layer`: The session layer
221///
222/// ### Returns
223/// - `Router`: The router that handles the web routes
224fn make_web_routes(
225    app_state: &handlers::AppState,
226    session_layer: SessionManagerLayer<MemoryStore>,
227) -> Router {
228    Router::new()
229        .merge(make_public_routes(app_state))
230        .merge(make_protected_routes(app_state))
231        .layer(axum::middleware::from_fn(
232            axum_tower_sessions_csrf::CsrfMiddleware::middleware,
233        ))
234        .layer(session_layer)
235        .layer(SetResponseHeaderLayer::if_not_present(
236            header::X_FRAME_OPTIONS,
237            HeaderValue::from_static("DENY"),
238        ))
239        .layer(SetResponseHeaderLayer::if_not_present(
240            header::X_CONTENT_TYPE_OPTIONS,
241            HeaderValue::from_static("nosniff"),
242        ))
243        .layer(SetResponseHeaderLayer::if_not_present(
244            header::HeaderName::from_static("x-xss-protection"),
245            HeaderValue::from_static("1; mode=block"),
246        ))
247        .layer(SetResponseHeaderLayer::if_not_present(
248            header::STRICT_TRANSPORT_SECURITY,
249            HeaderValue::from_static("max-age=31536000; includeSubDomains"),
250        ))
251        .layer(SetResponseHeaderLayer::if_not_present(
252            header::CONTENT_SECURITY_POLICY,
253            HeaderValue::from_static(
254                "default-src 'self'; script-src 'self' 'unsafe-inline' https://unpkg.com; style-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net;"
255            ),
256        ))
257}
