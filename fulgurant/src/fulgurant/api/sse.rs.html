36/// SSE endpoint handler for device connections
37///
38/// ### Arguments:
39/// - `state`: Application state containing SSE manager and configuration
40/// - `auth_user`: Authenticated user and device information from middleware
41///
42/// ### Returns:
43/// SSE stream that yields heartbeat and share_available events
44pub async fn handle_sse_connection(
45    State(state): State<AppState>,
46    Extension(auth_user): Extension<AuthenticatedUser>,
47) -> Sse<impl Stream<Item = Result<Event, Infallible>>> {
48    let device_id = auth_user.device_id.clone();
49    let device_name = auth_user.device_name.clone();
50    tracing::info!(
51        device_id = ?device_id,
52        device_name = ?device_name,
53        user_email = ?auth_user.user.email,
54        "SSE connection established"
55    );
56    let mut channel = state
57        .sse_manager
58        .create_channel([ChannelTag::DeviceId(device_id.clone())]);
59    let heartbeat_interval = Duration::from_secs(state.sse_heartbeat_seconds);
60    let stream = async_stream::stream! {
61        let mut interval = tokio::time::interval(heartbeat_interval);
62        loop {
63            tokio::select! {
64                _ = interval.tick() => {
65                    let timestamp = OffsetDateTime::now_utc()
66                        .format(&time::format_description::well_known::Rfc3339)
67                        .unwrap_or_default();
68                    let heartbeat = serde_json::json!({
69                        "timestamp": timestamp
70                    });
71                    tracing::debug!(device_id = ?device_id, "Sending heartbeat");
72                    yield Ok(Event::default()
73                        .event("heartbeat")
74                        .data(heartbeat.to_string()));
75                }
76                Some(notification) = channel.recv() => {
77                    tracing::info!(
78                        share_id = ?notification.share_id,
79                        device_id = ?device_id,
80                        file_name = ?notification.file_name,
81                        "Sending share notification via SSE"
82                    );
83                    match serde_json::to_string(&notification) {
84                        Ok(data) => {
85                            yield Ok(Event::default()
86                                .event("share_available")
87                                .data(data));
88                        }
89                        Err(e) => {
90                            tracing::error!(
91                                error = ?e,
92                                share_id = ?notification.share_id,
93                                "Failed to serialize share notification"
94                            );
95                        }
96                    }
97                }
98            }
99        }
100    };
101    Sse::new(stream).keep_alive(KeepAlive::default())
102}
