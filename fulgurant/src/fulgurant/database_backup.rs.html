6/// Get the backup folder path from environment or use default
7///
8/// ### Returns
9/// - `PathBuf`: Path to the backup folder
10pub fn get_backup_folder() -> PathBuf {
11    let backup_folder = std::env::var("BACKUP_FOLDER").unwrap_or_else(|_| "backups".to_string());
12    PathBuf::from(backup_folder)
13}

15/// Check if daily database backups are enabled
16pub fn is_daily_backup_enabled() -> bool {
17    std::env::var("DAILY_DATABASE_BACKUP")
18        .unwrap_or_else(|_| "false".to_string())
19        .to_lowercase()
20        == "true"
21}

23/// Validate that a path contains only safe characters for SQL interpolation
24///
25/// Since SQLite's VACUUM INTO doesn't support parameterized queries, we must
26/// sanitize the path manually to prevent SQL injection.
27///
28/// ### Arguments
29/// - `path`: The path string to validate
30///
31/// ### Returns
32/// - `Ok(())`: If the path is safe
33/// - `Err(anyhow::Error)`: If the path contains potentially dangerous characters
34fn validate_backup_path(path: &str) -> anyhow::Result<()> {
35    if path.contains('\'') {
36        return Err(anyhow::anyhow!(
37            "Backup path contains single quote character, which is not allowed for security reasons"
38        ));
39    }
40    Ok(())
41}

43/// Perform a database backup using SQLite's VACUUM INTO command
44///
45/// ### Arguments
46/// - `pool`: The SQLite connection pool
47///
48/// ### Returns
49/// - `Ok(PathBuf)`: Path to the created backup file
50/// - `Err(anyhow::Error)`: If backup fails
51pub async fn perform_backup(pool: &SqlitePool) -> anyhow::Result<PathBuf> {
52    let backup_folder = get_backup_folder();
53    std::fs::create_dir_all(&backup_folder)?;
54    let now = OffsetDateTime::now_utc();
55    let timestamp = now.format(&time::format_description::parse(
56        "[year]-[month]-[day]_[hour]-[minute]-[second]",
57    )?)?;
58    let backup_filename = format!("fulgurant_backup_{}.db", timestamp);
59    let backup_path = backup_folder.join(&backup_filename);
60    tracing::info!("Starting database backup to {}", backup_path.display());
61    let backup_path_str = backup_path
62        .to_str()
63        .ok_or_else(|| anyhow::anyhow!("Invalid backup path"))?;
64    validate_backup_path(backup_path_str)?;
65    let query = format!("VACUUM INTO '{}'", backup_path_str);
66    sqlx::query(&query).execute(pool).await?;
67    tracing::info!(
68        "Database backup completed successfully: {}",
69        backup_path.display()
70    );
71    Ok(backup_path)
72}

74/// Create a background task that performs daily database backups
75///
76/// ### Arguments
77/// - `pool`: The SQLite connection pool
78/// - `shutdown_token`: Token to signal graceful shutdown
79/// - `is_prod`: If true, performs final backup on shutdown (production). If false, skips final backup (development).
80pub fn make_daily_backup_task(pool: SqlitePool, shutdown_token: CancellationToken, is_prod: bool) {
81    if !is_daily_backup_enabled() {
82        tracing::info!("Daily database backups are disabled");
83        return;
84    }
85    let backup_folder = get_backup_folder();
86    tracing::info!(
87        "Starting daily database backup task (runs every 24 hours, backups stored in {})",
88        backup_folder.display()
89    );
90    tokio::spawn(async move {
91        let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(86400)); // 24 hours
92        loop {
93            tokio::select! {
94                _ = interval.tick() => {
95                    match perform_backup(&pool).await {
96                        Ok(backup_path) => {
97                            tracing::info!("Daily backup created: {}", backup_path.display());
98                        }
99                        Err(e) => {
100                            tracing::error!("Failed to create daily backup: {:?}", e);
101                        }
102                    }
103                },
104                _ = shutdown_token.cancelled() => {
105                    if is_prod {
106                        tracing::info!("Database backup task shutting down - performing final backup");
107                        match perform_backup(&pool).await {
108                            Ok(backup_path) => {
109                                tracing::info!("Final shutdown backup created: {}", backup_path.display());
110                            }
111                            Err(e) => {
112                                tracing::error!("Failed to create final shutdown backup: {:?}", e);
113                            }
114                        }
115                    } else {
116                        tracing::debug!("Database backup task shutting down (dev mode - skipping final backup)");
117                    }
118                    break;
119                }
120            }
121        }
122    });
123}

129    #[test]
130    fn test_validate_backup_path_rejects_single_quote() {
131        let malicious_path = "backups'; DROP TABLE users; --/backup.db";
132        let result = validate_backup_path(malicious_path);
133        assert!(result.is_err());
134        assert!(
135            result
136                .unwrap_err()
137                .to_string()
138                .contains("single quote character")
139        );
140    }

142    #[test]
143    fn test_validate_backup_path_allows_safe_paths() {
144        let safe_paths = vec![
145            "backups/fulgurant_backup_2026-02-15_12-00-00.db",
146            "/var/backups/fulgurant.db",
147            "./backups/test.db",
148            "../backups/backup.db",
149            "C:\\backups\\fulgurant.db",
150        ];

152        for path in safe_paths {
