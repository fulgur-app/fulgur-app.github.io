23/// GET /setup - Returns the initial setup page
24///
25/// ### Arguments
26/// - `state`: The state of the application
27/// - `session`: The session
28///
29/// ### Returns
30/// - `Ok(Html<String>)`: The setup page
31/// - `Err(AppError)`: An error occurred while rendering the template
32pub async fn get_setup_page(
33    State(state): State<AppState>,
34    session: Session,
35) -> Result<Response, AppError> {
36    let has_admin = state.user_repository.has_admin().await.map_err(|e| {
37        AppError::InternalError(anyhow::anyhow!("Failed to check for admin: {}", e))
38    })?;
39    if has_admin {
40        tracing::warn!(
41            "Attempted access to /setup but admin already exists - redirecting to /login"
42        );
43        return Ok(Redirect::to("/login").into_response());
44    }
45    let csrf_token = axum_tower_sessions_csrf::get_or_create_token(&session)
46        .await
47        .map_err(|e| {
48            AppError::InternalError(anyhow::anyhow!("Failed to generate CSRF token: {}", e))
49        })?;
50    let template = templates::SetupTemplate {
51        error_message: String::new(),
52        email: String::new(),
53        first_name: String::new(),
54        last_name: String::new(),
55        csrf_token,
56    };
57    Ok(Html(template.render()?).into_response())
58}

60/// POST /setup - Creates the initial admin user
61///
62/// ### Arguments
63/// - `state`: The state of the application
64/// - `session`: The session
65/// - `request`: The setup request
66///
67/// ### Returns
68/// - `Ok(Response)`: Redirect to index page after successful setup
69/// - `Err(AppError)`: An error occurred while creating the admin user
70pub async fn create_admin(
71    State(state): State<AppState>,
72    session: Session,
73    Form(request): Form<SetupRequest>,
74) -> Result<Response, AppError> {
75    let csrf_token = axum_tower_sessions_csrf::get_or_create_token(&session)
76        .await
77        .map_err(|e| {
78            AppError::InternalError(anyhow::anyhow!("Failed to generate CSRF token: {}", e))
79        })?;
80
81    let has_admin = state.user_repository.has_admin().await.map_err(|e| {
82        AppError::InternalError(anyhow::anyhow!("Failed to check for admin: {}", e))
83    })?;
84    if has_admin {
85        tracing::warn!(
86            "Attempted to create admin via /setup but admin already exists - rejecting request"
87        );
88        let mut response = Response::builder()
89            .status(StatusCode::OK)
90            .body("".to_string())
91            .map_err(|e| {
92                tracing::error!("Failed to build response: {}", e);
93                AppError::InternalError(anyhow::anyhow!("Failed to build response"))
94            })?;
95        response
96            .headers_mut()
97            .insert("HX-Redirect", HeaderValue::from_static("/login"));
98        return Ok(response.into_response());
99    }
100    let first_name = request.first_name.trim();
101    let last_name = request.last_name.trim();
102    let password = request.password.trim();
103    let email = request.email.trim().to_lowercase();
104    if !is_valid_email(&email) {
105        let template = templates::SetupFormTemplate {
106            error_message: "Invalid email address".to_string(),
107            email: email.clone(),
108            first_name: first_name.to_string(),
109            last_name: last_name.to_string(),
110            csrf_token,
111        };
112        return Ok(Html(template.render()?).into_response());
113    }
114    if !is_password_valid(password) {
115        let template = templates::SetupFormTemplate {
116            error_message: "Password must be 8-64 characters long and contain at least one uppercase letter, one lowercase letter, one digit, and one special character".to_string(),
117            email: email.clone(),
118            first_name: first_name.to_string(),
119            last_name: last_name.to_string(),
120            csrf_token,
121        };
122        return Ok(Html(template.render()?).into_response());
123    }
124    if first_name.is_empty() || last_name.is_empty() {
125        let template = templates::SetupFormTemplate {
126            error_message: "First name and last name are required".to_string(),
127            email: email.clone(),
128            first_name: first_name.to_string(),
129            last_name: last_name.to_string(),
130            csrf_token,
131        };
132        return Ok(Html(template.render()?).into_response());
133    }
134    let password_hash = hash_password(password)
135        .map_err(|e| AppError::InternalError(anyhow::anyhow!("Failed to hash password: {}", e)))?;
136    let user_id = state
137        .user_repository
138        .create_admin(
139            email.clone(),
140            first_name.to_string(),
141            last_name.to_string(),
142            password_hash,
143        )
144        .await
145        .map_err(|e| {
146            if matches!(e, sqlx::Error::RowNotFound) {
147                tracing::warn!("Admin already exists, rejecting duplicate creation");
148                AppError::InternalError(anyhow::anyhow!("Admin user already exists"))
149            } else {
150                tracing::error!("Failed to create admin user: {}", e);
151                AppError::InternalError(anyhow::anyhow!("Failed to create admin user: {}", e))
152            }
153        })?;
154    tracing::info!("Initial admin user created with ID: {}", user_id);
155    session
156        .insert(session::SESSION_USER_ID, user_id)
157        .await
158        .map_err(|e| {
159            AppError::InternalError(anyhow::anyhow!("Failed to set user id in session: {}", e))
160        })?;
161    state
162        .setup_needed
163        .store(false, std::sync::atomic::Ordering::Relaxed);
164    let mut response = Response::builder()
165        .status(StatusCode::OK)
166        .body("".to_string())
167        .map_err(|e| {
168            tracing::error!("Failed to build response: {}", e);
169            AppError::InternalError(anyhow::anyhow!("Failed to build response"))
170        })?;
171    response
172        .headers_mut()
173        .insert("HX-Redirect", HeaderValue::from_static("/"));
174    Ok(response.into_response())
175}
