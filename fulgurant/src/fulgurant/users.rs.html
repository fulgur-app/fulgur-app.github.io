10/// Escapes SQLite LIKE wildcards (`%`, `_`, `\`) in a search term.
11/// Must be used together with `LIKE ? ESCAPE '\'` in the SQL clause.
12fn escape_like(s: &str) -> String {
13    s.replace('\\', "\\\\")
14        .replace('%', "\\%")
15        .replace('_', "\\_")
16}

378    /// Create a new admin user (for initial setup, no email verification required)
379    ///
380    /// ### Arguments
381    /// - `email`: The email of the user
382    /// - `first_name`: The first name of the user
383    /// - `last_name`: The last name of the user
384    /// - `password_hash`: The password hash of the user
385    ///
386    /// ### Returns
387    /// - `Ok(i32)`: The ID of the created admin user
388    /// - `Err(sqlx::Error)`: The error if the operation fails
389    pub async fn create_admin(
390        &self,
391        email: String,
392        first_name: String,
393        last_name: String,
394        password_hash: String,
395    ) -> Result<i32, sqlx::Error> {
396        // Use transaction to prevent race condition
397        let mut tx = self.pool.begin().await?;
398        let count: (i32,) = sqlx::query_as("SELECT COUNT(*) FROM users WHERE role = 'Admin'")
399            .fetch_one(&mut *tx)
400            .await?;
401        if count.0 > 0 {
402            tx.rollback().await?;
403            return Err(sqlx::Error::RowNotFound);
404        }
405        let encryption_key = generate_encryption_key();
406        let result = sqlx::query(
407            "INSERT INTO users (email, first_name, last_name, password_hash, role, encryption_key, email_verified) VALUES (?, ?, ?, ?, 'Admin', ?, TRUE)",
408        )
409        .bind(email)
410        .bind(first_name)
411        .bind(last_name)
412        .bind(password_hash)
413        .bind(encryption_key)
414        .execute(&mut *tx)
415        .await?;
416        tx.commit().await?;
417        let id = result.last_insert_rowid() as i32;
418        Ok(id)
419    }

464    /// Search users with filters and pagination
465    ///
466    /// ### Arguments
467    /// - `email`: Optional email filter (contains search, case-insensitive)
468    /// - `first_name`: Optional first name filter (contains search, case-insensitive)
469    /// - `last_name`: Optional last name filter (contains search, case-insensitive)
470    /// - `role`: Optional role filter (exact match, or "All" for no filter)
471    /// - `page`: The page number (1-indexed)
472    /// - `page_size`: The number of users per page
473    ///
474    /// ### Returns
475    /// - `Ok(PaginatedUsers)`: The paginated list of filtered users (without sensitive information)
476    /// - `Err(sqlx::Error)`: The error if the operation fails
477    pub async fn search(
478        &self,
479        email: Option<String>,
480        first_name: Option<String>,
481        last_name: Option<String>,
482        role: Option<String>,
483        page: i32,
484        page_size: i32,
485    ) -> Result<PaginatedUsers, sqlx::Error> {
486        let page = page.max(1);
487        let page_size = page_size.max(1);
488        let offset = (page - 1) * page_size;
489        let mut where_clauses = Vec::new();
490        let mut params = Vec::new();
491        if let Some(e) = email.as_ref().filter(|s| !s.trim().is_empty()) {
492            where_clauses.push("email LIKE ? ESCAPE '\\'");
493            params.push(format!("%{}%", escape_like(e.trim())));
494        }
495        if let Some(f) = first_name.as_ref().filter(|s| !s.trim().is_empty()) {
496            where_clauses.push("first_name LIKE ? ESCAPE '\\'");
497            params.push(format!("%{}%", escape_like(f.trim())));
498        }
499        if let Some(l) = last_name.as_ref().filter(|s| !s.trim().is_empty()) {
500            where_clauses.push("last_name LIKE ? ESCAPE '\\'");
501            params.push(format!("%{}%", escape_like(l.trim())));
502        }
503        if let Some(r) = role
504            .as_ref()
505            .filter(|s| !s.trim().is_empty() && *s != "All")
506        {
507            where_clauses.push("role = ?");
508            params.push(r.trim().to_string());
509        }
510        let where_clause = if where_clauses.is_empty() {
511            String::new()
512        } else {
513            format!("WHERE {}", where_clauses.join(" AND "))
514        };
515        let count_query = format!("SELECT COUNT(*) FROM users {}", where_clause);
516        let mut count_query_builder = sqlx::query_as::<_, (i32,)>(&count_query);
517        for param in &params {
518            count_query_builder = count_query_builder.bind(param);
519        }
520        let total_count = count_query_builder.fetch_one(&self.pool).await?.0;
521
522        let total_pages = (total_count + page_size - 1) / page_size;
523        let query = format!(
524            "SELECT id, email, first_name, last_name, email_verified, role, last_activity, shares, force_password_update, created_at, updated_at FROM users {} ORDER BY created_at DESC LIMIT ? OFFSET ?",
525            where_clause
526        );
527        let mut query_builder = sqlx::query_as::<_, DisplayUser>(&query);
528        for param in &params {
529            query_builder = query_builder.bind(param);
530        }
531        query_builder = query_builder.bind(page_size).bind(offset);
532        let users = query_builder.fetch_all(&self.pool).await?;
533        Ok(PaginatedUsers {
534            users,
535            total_count,
536            page,
537            page_size,
538            total_pages,
539        })
540    }
